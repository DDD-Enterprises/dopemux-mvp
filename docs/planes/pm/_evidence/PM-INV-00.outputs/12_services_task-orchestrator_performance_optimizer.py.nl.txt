   186	            logger.info(f"Learned {len(valid_patterns)} patterns for user {user_id}")

   296	    async def _load_user_patterns(self) -> None:
   297	        """Load existing productivity patterns from storage."""
   298	        loaded_patterns = 0
   299	        raw_payload: Any = None
   300	
   301	        if self.conport:
   302	            for method_name in ("get_productivity_patterns", "get_optimization_patterns", "query_productivity_patterns"):
   303	                if not hasattr(self.conport, method_name):
   304	                    continue
   305	                try:
   306	                    result = getattr(self.conport, method_name)()
   307	                    if asyncio.iscoroutine(result):
   308	                        result = await result
   309	                    raw_payload = result
   310	                    break
   311	                except Exception as exc:
   312	                    logger.debug("Pattern load via %s failed: %s", method_name, exc)
   313	
   314	        if raw_payload is None:
   315	            self.productivity_patterns.setdefault("default", [])
   316	            return
   317	
   318	        if isinstance(raw_payload, dict):
   319	            # Expected shape: {user_id: [pattern_dicts]}
   320	            iterable = raw_payload.items()
   321	        elif isinstance(raw_payload, list):
   322	            # Flat shape: [{user_id: ..., ...}]
   323	            grouped: Dict[str, List[Dict[str, Any]]] = {}
   324	            for item in raw_payload:
   325	                if not isinstance(item, dict):
   326	                    continue
   327	                user_id = str(item.get("user_id", "default"))
   328	                grouped.setdefault(user_id, []).append(item)
   329	            iterable = grouped.items()
   330	        else:
   331	            iterable = []
   332	
   333	        for user_id, patterns in iterable:
   334	            parsed: List[ProductivityPattern] = []
   335	            for item in patterns or []:
   336	                if not isinstance(item, dict):
   337	                    continue
   338	                try:
   339	                    pattern = ProductivityPattern(
   340	                        pattern_id=str(item.get("pattern_id", f"pattern_{uuid.uuid4().hex[:8]}")),
   341	                        user_id=str(item.get("user_id", user_id)),
   342	                        pattern_type=str(item.get("pattern_type", "unknown")),
   343	                        confidence=float(item.get("confidence", 0.6)),
   344	                        effectiveness_score=float(item.get("effectiveness_score", 0.6)),
   345	                        optimal_times=[time.fromisoformat(t) if isinstance(t, str) else t for t in item.get("optimal_times", [])],
   346	                        duration_range=tuple(item.get("duration_range", (25.0, 90.0))),  # type: ignore[arg-type]
   347	                        prerequisites=list(item.get("prerequisites", [])),
   348	                        success_indicators=list(item.get("success_indicators", [])),
   349	                        adhd_accommodations=list(item.get("adhd_accommodations", [])),
   350	                        energy_requirements=list(item.get("energy_requirements", [])),
   351	                        attention_requirements=list(item.get("attention_requirements", [])),
   352	                        data_points=int(item.get("data_points", 0)),
   353	                        last_updated=datetime.fromisoformat(item.get("last_updated")) if isinstance(item.get("last_updated"), str) else datetime.now(),
   354	                        trend_direction=str(item.get("trend_direction", "stable")),
   355	                    )
   356	                    parsed.append(pattern)
   357	                    loaded_patterns += 1
   358	                except Exception as exc:
   359	                    logger.debug("Failed parsing productivity pattern for %s: %s", user_id, exc)
   360	            self.productivity_patterns[str(user_id)] = parsed
   361	
   362	        logger.info("Loaded %d productivity patterns", loaded_patterns)

   375	        """Start background performance monitoring."""

   380	    async def _performance_monitor_loop(self) -> None:
   381	        """Background loop to track drift and model health."""
   382	        try:
   383	            while True:
   384	                await asyncio.sleep(300)
   385	                total_patterns = sum(len(v) for v in self.productivity_patterns.values())
   386	                if total_patterns == 0:
   387	                    continue
   388	
   389	                confidences = [
   390	                    p.confidence
   391	                    for patterns in self.productivity_patterns.values()
   392	                    for p in patterns
   393	                ]
   394	                avg_confidence = statistics.mean(confidences) if confidences else 0.0
   395	
   396	                if avg_confidence < 0.5:
   397	                    logger.info("Pattern confidence low (%.2f), increasing exploration", avg_confidence)
   398	                    self.learning_rate = min(0.3, self.learning_rate + 0.02)
   399	                else:
   400	                    self.learning_rate = max(0.05, self.learning_rate - 0.01)
   401	        except asyncio.CancelledError:
   402	            logger.debug("Performance monitor loop cancelled")
   403	            raise
   404	        except Exception as exc:
   405	            logger.error("Performance monitor loop failed: %s", exc)

   585	        """Apply ADHD-specific optimizations to workflow."""
   586	        adhd_optimized = []
   587	
   588	        # Group similar tasks to reduce context switching
   589	        task_groups = self._group_similar_tasks(tasks)
   590	
   591	        for group in task_groups:

   605	        """Group similar tasks to minimize context switching."""
   606	        # Simple grouping by task type
   607	        groups: Dict[str, List[Dict[str, Any]]] = {}

   609	        for task in tasks:

   644	    async def _estimate_workflow_improvement(
   645	        self,
   646	        original: List[Dict[str, Any]],
   647	        optimized: List[Dict[str, Any]],
   648	        patterns: List[ProductivityPattern]
   649	    ) -> Dict[str, Any]:
   650	        """Estimate improvement from workflow optimization."""
   651	        # Simplified improvement calculation
   652	        context_switches_original = self._count_context_switches(original)
   653	        context_switches_optimized = self._count_context_switches(optimized)
   654	
   655	        context_switch_improvement = (
   656	            context_switches_original - context_switches_optimized
   657	        ) / max(context_switches_original, 1)
   658	
   659	        return {
   660	            "productivity_gain": min(0.5, context_switch_improvement * 0.3),  # Cap at 50%
   661	            "context_switches_reduced": context_switches_original - context_switches_optimized,
   662	            "adhd_accommodations": [f"Accommodation for task {i}" for i in range(len(optimized))]
   663	        }
   664	
   665	    def _count_context_switches(self, tasks: List[Dict[str, Any]]) -> int:
   666	        """Count context switches in task workflow."""
   667	        if len(tasks) <= 1:
   668	            return 0
   669	
   670	        switches = 0
   671	        for i in range(1, len(tasks)):
   672	            if self._requires_context_switch(tasks[i-1], tasks[i]):
   673	                switches += 1
   674	
   675	        return switches

   677	    async def _update_optimization_models(
   678	        self, recommendation_id: str, results: Dict[str, Any]
   679	    ) -> None:
   680	        """Update optimization models based on results."""
   681	        strategy_hint = str(results.get("strategy", "")).lower()
   682	        if not strategy_hint:
   683	            for strategy in OptimizationStrategy:
   684	                if strategy.value in recommendation_id:
   685	                    strategy_hint = strategy.value
   686	                    break
   687	
   688	        target_strategy = None
   689	        for strategy in OptimizationStrategy:
   690	            if strategy.value == strategy_hint:
   691	                target_strategy = strategy
   692	                break
   693	
   694	        if target_strategy is None:
   695	            target_strategy = OptimizationStrategy.TASK_REORDERING
   696	
   697	        model = self.optimization_models.setdefault(
   698	            target_strategy,
   699	            {
   700	                "effectiveness_score": 0.6,
   701	                "confidence": 0.5,
   702	                "usage_count": 0,
   703	                "success_rate": 0.5,
   704	            },
   705	        )
   706	
   707	        improvement = float(results.get("productivity_improvement", 0.0))
   708	        success = improvement > 0.05
   709	        confidence = float(results.get("confidence", 0.7))
   710	
   711	        model["usage_count"] += 1
   712	        old_success = float(model.get("success_rate", 0.5))
   713	        old_effectiveness = float(model.get("effectiveness_score", 0.6))
   714	        old_confidence = float(model.get("confidence", 0.5))
   715	
   716	        outcome_value = 1.0 if success else 0.0
   717	        model["success_rate"] = round((1 - self.learning_rate) * old_success + self.learning_rate * outcome_value, 4)
   718	        model["effectiveness_score"] = round(
   719	            (1 - self.learning_rate) * old_effectiveness + self.learning_rate * max(0.0, min(1.0, 0.5 + improvement)),
   720	            4,
   721	        )
   722	        model["confidence"] = round((1 - self.learning_rate) * old_confidence + self.learning_rate * confidence, 4)
   723	
   724	    async def close(self) -> None:
   725	        """Shutdown background optimizer tasks."""
   726	        if self._monitor_task and not self._monitor_task.done():
   727	            self._monitor_task.cancel()
   728	            await asyncio.gather(self._monitor_task, return_exceptions=True)
