   164	    async def orchestrate_deployment(
   165	        self,
   166	        pipeline_id: str,
   167	        git_branch: str = "main",
   168	        adhd_mode: bool = True
   169	    ) -> Optional[str]:
   170	        """Orchestrate a complete deployment pipeline."""
   171	        try:
   172	            pipeline = self.pipelines.get(pipeline_id)
   173	            if not pipeline:
   174	                logger.error(f"Pipeline {pipeline_id} not found")
   175	                return None
   176	
   177	            # Create deployment instance
   178	            deployment_instance_id = f"deploy_{uuid.uuid4().hex[:8]}"
   179	            deployment_instance = self._create_deployment_instance(pipeline, git_branch)
   180	
   181	            # Apply ADHD optimizations if requested
   182	            if adhd_mode:
   183	                deployment_instance = await self._apply_adhd_deployment_optimizations(deployment_instance)
   184	
   185	            # Start deployment
   186	            self.active_deployments[deployment_instance_id] = deployment_instance
   187	
   188	            # Create monitoring task
   189	            monitor_task = asyncio.create_task(
   190	                self._monitor_deployment(deployment_instance_id, deployment_instance)
   191	            )
   192	            self.monitoring_tasks[deployment_instance_id] = monitor_task
   193	
   194	            # Execute deployment pipeline
   195	            success = await self._execute_deployment_pipeline(deployment_instance)
   196	
   197	            self.deployment_stats["deployments_orchestrated"] += 1
   198	            if success:
   199	                self.deployment_stats["successful_deployments"] += 1
   200	
   201	            logger.info(f"Deployment {deployment_instance_id} {'succeeded' if success else 'failed'}")
   202	            return deployment_instance_id
   203	
   204	        except Exception as e:
   205	            logger.error(f"Deployment orchestration failed: {e}")
   206	            return None

   421	    async def _execute_deployment_task(self, task: DeploymentTask) -> bool:
   422	        """Execute individual deployment task."""
   423	        try:
   424	            task.status = DeploymentStatus.RUNNING
   425	            task.started_at = datetime.now()
   426	
   427	            logger.info(f"⚡ Executing: {task.name}")
   428	
   429	            # Execute command with timeout
   430	            process = await asyncio.create_subprocess_shell(
   431	                task.command,
   432	                stdout=asyncio.subprocess.PIPE,
   433	                stderr=asyncio.subprocess.PIPE
   434	            )
   435	
   436	            try:
   437	                stdout, stderr = await asyncio.wait_for(
   438	                    process.communicate(),
   439	                    timeout=task.timeout_minutes * 60
   440	                )
   441	
   442	                if process.returncode == 0:
   443	                    task.status = DeploymentStatus.SUCCESS
   444	                    task.output_log.append(stdout.decode() if stdout else "")
   445	                    return True
   446	                else:
   447	                    task.status = DeploymentStatus.FAILED
   448	                    task.error_message = stderr.decode() if stderr else "Unknown error"
   449	                    return False
   450	
   451	            except asyncio.TimeoutError:
   452	                process.kill()
   453	                task.status = DeploymentStatus.FAILED
   454	                task.error_message = f"Timeout after {task.timeout_minutes} minutes"
   455	                return False
   456	
   457	        except Exception as e:
   458	            task.status = DeploymentStatus.FAILED
   459	            task.error_message = str(e)
   460	            return False
   461	        finally:
   462	            task.completed_at = datetime.now()

   531	    async def _simplify_error_message(self, error_text: str) -> str:
   532	        """Simplify error messages for ADHD users."""
   533	        # Extract key information from complex error messages
   534	        lines = error_text.split('\n')
   535	
   536	        # Look for common patterns
   537	        for line in lines:
   538	            if "FAILED" in line and "::" in line:
   539	                return f"Test failed: {line.split('::')[-1].strip()}"
   540	            elif "Error:" in line:
   541	                return line.strip()
   542	            elif "AssertionError" in line:
   543	                return f"Assertion failed: {line.split('AssertionError:')[-1].strip()}"
   544	
   545	        # Return first non-empty line if no patterns found
   546	        for line in lines:
   547	            if line.strip():
   548	                return line.strip()[:100] + "..." if len(line) > 100 else line.strip()
   549	
   550	        return "Unknown error occurred"

   552	    def _get_current_stage(self, deployment: DeploymentPipeline) -> str:
   553	        """Get current deployment stage."""
   554	        # Find first non-completed task
   555	        for task in deployment.tasks:
   556	            if task.status in [DeploymentStatus.PENDING, DeploymentStatus.RUNNING]:
   557	                return task.stage.value
   558	
   559	        return "completed"

   561	    def _calculate_progress(self, deployment: DeploymentPipeline) -> float:
   562	        """Calculate deployment progress percentage."""
   563	        if not deployment.tasks:
   564	            return 0.0
   565	
   566	        completed_tasks = sum(1 for task in deployment.tasks
   567	                            if task.status == DeploymentStatus.SUCCESS)
   568	        return (completed_tasks / len(deployment.tasks)) * 100

   570	    def _estimate_completion_time(self, deployment: DeploymentPipeline) -> Optional[datetime]:
   571	        """Estimate deployment completion time."""
   572	        remaining_tasks = [task for task in deployment.tasks
   573	                         if task.status not in [DeploymentStatus.SUCCESS, DeploymentStatus.FAILED]]
   574	
   575	        if not remaining_tasks:
   576	            return datetime.now()  # Already complete
   577	
   578	        remaining_time = sum(task.timeout_minutes for task in remaining_tasks)
   579	        return datetime.now() + timedelta(minutes=remaining_time)

   581	    async def _get_health_indicators(self, deployment: DeploymentPipeline) -> List[str]:
   582	        """Get deployment health indicators."""
   583	        indicators = []
   584	
   585	        # Check task success rate
   586	        completed_tasks = [t for t in deployment.tasks if t.status != DeploymentStatus.PENDING]
   587	        if completed_tasks:
   588	            success_rate = sum(1 for t in completed_tasks if t.status == DeploymentStatus.SUCCESS) / len(completed_tasks)
   589	            if success_rate > 0.8:
   590	                indicators.append("✅ High success rate")
   591	            elif success_rate < 0.5:
   592	                indicators.append("⚠️ Low success rate")
   593	
   594	        # Check timing
   595	        current_time = datetime.now()
   596	        if deployment.created_at:
   597	            elapsed = (current_time - deployment.created_at).total_seconds() / 60
   598	            if elapsed > deployment.estimated_duration * 1.5:
   599	                indicators.append("⏰ Running over estimated time")
   600
