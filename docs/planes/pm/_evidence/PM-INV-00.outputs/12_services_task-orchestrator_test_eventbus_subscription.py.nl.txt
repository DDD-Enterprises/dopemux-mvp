     7	2. Event subscription
     8	3. Event publication

    10	"""
    11	
    12	import asyncio
    13	import sys
    14	import os
    15	import logging
    16	from datetime import datetime
    17	
    18	# Add DopeconBridge path
    19	dopecon_bridge_path = os.path.join(os.path.dirname(__file__), '..', 'mcp-dopecon-bridge')
    20	sys.path.insert(0, dopecon_bridge_path)

    22	from event_bus import EventBus, Event, EventType
    23	
    24	logging.basicConfig(level=logging.INFO, format='%(asctime)s - %(name)s - %(levelname)s - %(message)s')
    25	logger = logging.getLogger(__name__)
    26	
    27	
    28	class TestSubscriber:
    29	    """Test subscriber that listens for events."""
    30	
    31	    def __init__(self, redis_url: str = "redis://localhost:6379"):
    32	        self.redis_url = redis_url
    33	        self.event_bus: EventBus = None
    34	        self.received_events = []
    35	        self.running = False
    36	
    37	    async def initialize(self):
    38	        """Initialize EventBus."""
    39	        logger.info("🚀 Initializing EventBus...")
    40	        self.event_bus = EventBus(self.redis_url, password=None)
    41	        await self.event_bus.initialize()
    42	        logger.info("✅ EventBus initialized")
    43	
    44	    async def subscribe_and_listen(self, duration_seconds: int = 10):
    45	        """Subscribe to events for specified duration."""
    46	        logger.info(f"📡 Subscribing to dopemux:events stream for {duration_seconds} seconds...")
    47	
    48	        self.running = True
    49	
    50	        # Create subscription task
    51	        subscription_task = asyncio.create_task(self._subscription_worker())
    52	
    53	        # Wait for specified duration
    54	        await asyncio.sleep(duration_seconds)
    55	
    56	        # Stop subscription
    57	        self.running = False
    58	        subscription_task.cancel()
    59	
    60	        try:
    61	            await subscription_task
    62	        except asyncio.CancelledError:
    63	            logger.info("📭 Subscription stopped")
    64	
    65	        logger.info(f"📥 Received {len(self.received_events)} events total")
    66	
    67	        return self.received_events
    68	
    69	    async def _subscription_worker(self):
    70	        """Background worker for event subscription."""
    71	        try:
    72	            async for msg_id, event in self.event_bus.subscribe(
    73	                stream="dopemux:events",
    74	                consumer_group="test-subscriber",
    75	                consumer_name="test-instance-1"
    76	            ):
    77	                if not self.running:
    78	                    break
    79	
    80	                logger.info(f"📥 Received event: {event.type} from {event.source}")
    81	                logger.info(f"   Data: {event.data}")
    82	                self.received_events.append(event)
    83	
    84	        except Exception as e:
    85	            if self.running:  # Only log if not intentionally stopped
    86	                logger.error(f"Subscription error: {e}")
    87	
    88	    async def close(self):
    89	        """Close EventBus connection."""
    90	        if self.event_bus:
    91	            await self.event_bus.close()
    92	            logger.info("📪 EventBus closed")
    93

   144	async def main():
   145	    """Main test execution."""
   146	    logger.info("=" * 70)
   147	    logger.info("Component 3: EventBus Subscription Test")
   148	    logger.info("=" * 70)
   149	
   150	    # Create subscriber

   159	        await publisher_bus.initialize()

   168	
   169	        # Publish test events

   172	        # Wait for subscription to complete
   173	        received_events = await subscription_task

   175	        # Verify results
