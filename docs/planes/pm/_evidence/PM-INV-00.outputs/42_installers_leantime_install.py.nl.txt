    95	    async def run_installation(self, config_file: Optional[str] = None) -> bool:
    96	        """
    97	        Run the complete installation process.
    98	
    99	        Args:
   100	            config_file: Optional configuration file path
   101	
   102	        Returns:
   103	            True if installation successful
   104	        """
   105	        try:
   106	            logger.info("🚀 Starting Dopemux Leantime & Task-Master installation...")
   107	
   108	            # Load configuration
   109	            if config_file:
   110	                await self._load_config_file(config_file)
   111	            else:
   112	                await self._interactive_config()
   113	
   114	            # Pre-flight checks
   115	            self.log_step("Running pre-flight checks...")
   116	            if not await self._run_preflight_checks():
   117	                logger.error("❌ Pre-flight checks failed")
   118	                return False
   119	
   120	            # Generate secure passwords if not provided
   121	            await self._generate_secure_credentials()
   122	
   123	            # Install components
   124	            self.log_step("Installing Docker environment...")
   125	            if not await self._install_docker_environment():
   126	                logger.error("❌ Docker environment installation failed")
   127	                return False
   128	
   129	            self.log_step("Setting up Task-Master AI...")
   130	            if not await self._install_taskmaster():
   131	                logger.error("❌ Task-Master AI installation failed")
   132	                return False
   133	
   134	            self.log_step("Configuring integrations...")
   135	            if not await self._configure_integrations():
   136	                logger.error("❌ Integration configuration failed")
   137	                return False
   138	
   139	            # Start services
   140	            self.log_step("Starting services...")
   141	            if not await self._start_services():
   142	                logger.error("❌ Service startup failed")
   143	                return False
   144	
   145	            # Verify installation
   146	            self.log_step("Verifying installation...")
   147	            if not await self._verify_installation():
   148	                logger.error("❌ Installation verification failed")
   149	                return False
   150	
   151	            # Post-installation setup
   152	            await self._post_installation_setup()
   153	
   154	            logger.info("✅ Installation completed successfully!")
   155	            await self._display_installation_summary()
   156	
   157	            return True
   158	
   159	        except Exception as e:
   160	            logger.error(f"❌ Installation failed: {e}")
   161	            await self._cleanup_failed_installation()
   162	            return False

   169	    async def _load_config_file(self, config_file: str):
   170	        """Load configuration from file."""
   171	        config_path = Path(config_file)
   172	        if not config_path.exists():
   173	            raise FileNotFoundError(f"Configuration file not found: {config_file}")
   174	
   175	        with open(config_path, 'r') as f:
   176	            if config_path.suffix.lower() == '.yaml' or config_path.suffix.lower() == '.yml':
   177	                config_data = yaml.safe_load(f)
   178	            else:
   179	                config_data = json.load(f)
   180	
   181	        # Update install config
   182	        for key, value in config_data.items():
   183	            if hasattr(self.install_config, key):
   184	                setattr(self.install_config, key, value)

   258	    async def _run_preflight_checks(self) -> bool:
   259	        """Run comprehensive pre-flight system checks."""
   260	        checks_passed = True
   261	
   262	        # Check Docker
   263	        try:
   264	            self.docker_client = docker.from_env()
   265	            self.docker_client.ping()
   266	            self.preflight_results["docker"] = {"status": "ok", "version": self.docker_client.version()}
   267	            logger.info("✅ Docker is available and running")
   268	        except DockerException as e:
   269	            self.preflight_results["docker"] = {"status": "error", "error": str(e)}
   270	            logger.error("❌ Docker is not available or not running")
   271	            checks_passed = False
   272	
   273	        # Check Docker Compose
   274	        try:
   275	            result = subprocess.run(
   276	                ["docker-compose", "--version"],
   277	                capture_output=True, text=True, check=True
   278	            )
   279	            self.preflight_results["docker_compose"] = {"status": "ok", "version": result.stdout.strip()}
   280	            logger.info("✅ Docker Compose is available")
   281	        except (subprocess.CalledProcessError, FileNotFoundError):
   282	            self.preflight_results["docker_compose"] = {"status": "error", "error": "Not found"}
   283	            logger.error("❌ Docker Compose is not available")
   284	            checks_passed = False
   285	
   286	        # Check Node.js and npm for Task-Master
   287	        try:
   288	            node_result = subprocess.run(
   289	                ["node", "--version"],
   290	                capture_output=True, text=True, check=True
   291	            )
   292	            npm_result = subprocess.run(
   293	                ["npm", "--version"],
   294	                capture_output=True, text=True, check=True
   295	            )
   296	            self.preflight_results["nodejs"] = {
   297	                "status": "ok",
   298	                "node_version": node_result.stdout.strip(),
   299	                "npm_version": npm_result.stdout.strip()
   300	            }
   301	            logger.info("✅ Node.js and npm are available")
   302	        except (subprocess.CalledProcessError, FileNotFoundError):
   303	            self.preflight_results["nodejs"] = {"status": "error", "error": "Node.js or npm not found"}
   304	            logger.error("❌ Node.js or npm is not available")
   305	            if self.install_config.taskmaster_enabled:
   306	                checks_passed = False
   307	
   308	        # Check available ports
   309	        ports_to_check = [
   310	            self.install_config.leantime_port,
   311	            self.install_config.mysql_port,
   312	            self.install_config.redis_port
   313	        ]
   314	
   315	        available_ports = []
   316	        for port in ports_to_check:
   317	            if await self._check_port_available(port):
   318	                available_ports.append(port)
   319	                logger.info(f"✅ Port {port} is available")
   320	            else:
   321	                logger.warning(f"⚠️ Port {port} is in use")
   322	
   323	        self.preflight_results["ports"] = {
   324	            "requested": ports_to_check,
   325	            "available": available_ports
   326	        }
   327	
   328	        # Check disk space (require at least 2GB)
   329	        try:
   330	            disk_usage = shutil.disk_usage(self.installation_dir)
   331	            free_gb = disk_usage.free / (1024**3)
   332	            self.preflight_results["disk_space"] = {"free_gb": free_gb, "status": "ok" if free_gb >= 2 else "warning"}
   333	
   334	            if free_gb < 2:
   335	                logger.warning(f"⚠️ Only {free_gb:.1f}GB free disk space (recommended: 2GB+)")
   336	            else:
   337	                logger.info(f"✅ Sufficient disk space: {free_gb:.1f}GB available")
   338	        except Exception as e:
   339	            self.preflight_results["disk_space"] = {"status": "error", "error": str(e)}
   340	
   341	        # Check Python dependencies
   342	        try:
   343	            import yaml, click
   344	            self.preflight_results["python_deps"] = {"status": "ok"}
   345	            logger.info("✅ Required Python dependencies are available")
   346	        except ImportError as e:
   347	            self.preflight_results["python_deps"] = {"status": "error", "error": str(e)}
   348	            logger.error(f"❌ Missing Python dependencies: {e}")
   349	            checks_passed = False
   350	
   351	        return checks_passed

   353	    async def _check_port_available(self, port: int) -> bool:
   354	        """Check if a port is available."""
   355	        import socket
   356	
   357	        try:
   358	            with socket.socket(socket.AF_INET, socket.SOCK_STREAM) as sock:
   359	                sock.bind(('localhost', port))
   360	                return True
   361	        except OSError:
   362	            return False

   364	    async def _generate_secure_credentials(self):
   365	        """Generate secure passwords and tokens if not provided."""
   366	        def generate_password(length: int = 32) -> str:
   367	            alphabet = string.ascii_letters + string.digits + "!@#$%^&*"
   368	            return ''.join(secrets.choice(alphabet) for _ in range(length))
   369	
   370	        if not self.install_config.mysql_root_password:
   371	            self.install_config.mysql_root_password = generate_password()
   372	
   373	        if not self.install_config.mysql_password:
   374	            self.install_config.mysql_password = generate_password()
   375	
   376	        if not self.install_config.lean_session_password:
   377	            self.install_config.lean_session_password = generate_password(64)
   378	
   379	        if not self.install_config.redis_password:
   380	            self.install_config.redis_password = generate_password()
   381	
   382	        logger.info("🔐 Generated secure credentials")

   398	
   399	            # Copy docker-compose.yml if it doesn't exist
   400	            compose_file = self.docker_dir / "docker-compose.yml"
   401	            if not compose_file.exists():
   402	                # Use the existing docker-compose.yml we created earlier
   403	                source_compose = Path(__file__).parent.parent.parent / "docker" / "leantime" / "docker-compose.yml"

   405	                    shutil.copy2(source_compose, compose_file)
   406	                    logger.info("✅ Docker Compose configuration ready")
   407	                else:
   408	                    logger.error("❌ Docker Compose template not found")
   409	                    return False
   410	
   411	            return True
   412	
   413	        except Exception as e:
   414	            logger.error(f"❌ Docker environment setup failed: {e}")
   415	            return False

   467	    async def _install_taskmaster(self) -> bool:
   468	        """Install and configure Task-Master AI."""
   469	        if not self.install_config.taskmaster_enabled:
   470	            logger.info("⏭️ Skipping Task-Master AI installation (disabled)")
   471	            return True
   472	
   473	        try:
   474	            # Install task-master-ai globally
   475	            logger.info("📦 Installing task-master-ai package...")
   476	
   477	            result = await asyncio.create_subprocess_exec(
   478	                "npm", "install", "-g", "task-master-ai",
   479	                stdout=asyncio.subprocess.PIPE,
   480	                stderr=asyncio.subprocess.PIPE
   481	            )
   482	
   483	            stdout, stderr = await result.communicate()
   484	
   485	            if result.returncode != 0:
   486	                logger.error(f"❌ npm install failed: {stderr.decode()}")
   487	                return False
   488	
   489	            logger.info("✅ Task-Master AI installed successfully")
   490	
   491	            # Create Task-Master configuration
   492	            await self._create_taskmaster_config()
   493	
   494	            return True
   495	
   496	        except Exception as e:
   497	            logger.error(f"❌ Task-Master installation failed: {e}")
   498	            return False

   516	            "openai": "gpt-4",
   517	            "google": "gemini-pro",
   518	            "perplexity": "llama-3.1-sonar-small-128k-online",
   519	            "mistral": "mistral-large-latest",
   520	            "groq": "mixtral-8x7b-32768",
   521	            "ollama": "llama2"
   522	        }

   533	            config_data["defaultProvider"] = list(config_data["providers"].keys())[0]

   543	        tasks_file = tasks_dir / "tasks.json"
   544	        with open(tasks_file, 'w') as f:
   545	            json.dump({"tasks": [], "metadata": {}}, f, indent=2)

   586	        """Start Docker services."""
   587	        try:
   588	            # Change to Docker directory
   589	            original_cwd = os.getcwd()
   590	            os.chdir(self.docker_dir)
   591	
   592	            try:
   593	                # Pull images first
   594	                logger.info("📥 Pulling Docker images...")
   595	                result = await asyncio.create_subprocess_exec(
   596	                    "docker-compose", "pull",
   597	                    stdout=asyncio.subprocess.PIPE,
   598	                    stderr=asyncio.subprocess.PIPE
   599	                )
   600	                await result.communicate()
   601	
   602	                # Start services
   603	                logger.info("🚀 Starting Docker services...")
   604	                result = await asyncio.create_subprocess_exec(
   605	                    "docker-compose", "up", "-d",
   606	                    stdout=asyncio.subprocess.PIPE,
   607	                    stderr=asyncio.subprocess.PIPE
   608	                )
   609	
   610	                stdout, stderr = await result.communicate()
   611	
   612	                if result.returncode != 0:
   613	                    logger.error(f"❌ Docker Compose failed: {stderr.decode()}")
   614	                    return False
   615	
   616	                logger.info("✅ Docker services started successfully")
   617	
   618	                # Wait for services to be ready
   619	                await self._wait_for_services()
   620	
   621	                return True
   622	
   623	            finally:
   624	                os.chdir(original_cwd)
   625	
   626	        except Exception as e:
   627	            logger.error(f"❌ Service startup failed: {e}")
   628	            return False

   630	    async def _wait_for_services(self):
   631	        """Wait for services to be ready."""
   632	        import aiohttp
   633	
   634	        max_attempts = 30
   635	        wait_interval = 10
   636	
   637	        logger.info("⏳ Waiting for services to be ready...")
   638	
   639	        for attempt in range(max_attempts):
   640	            try:
   641	                # Check Leantime health
   642	                async with aiohttp.ClientSession() as session:
   643	                    async with session.get(
   644	                        f"http://localhost:{self.install_config.leantime_port}/install",
   645	                        timeout=aiohttp.ClientTimeout(total=5)
   646	                    ) as response:
   647	                        if response.status in [200, 302]:  # 302 for redirect to installation
   648	                            logger.info("✅ Leantime is ready")
   649	                            return
   650	
   651	            except Exception:
   652	                pass
   653	
   654	            if attempt < max_attempts - 1:
   655	                logger.info(f"⏳ Services not ready yet, waiting {wait_interval}s... (attempt {attempt + 1}/{max_attempts})")
   656	                await asyncio.sleep(wait_interval)
   657	
   658	        logger.warning("⚠️ Services may not be fully ready, but continuing...")
