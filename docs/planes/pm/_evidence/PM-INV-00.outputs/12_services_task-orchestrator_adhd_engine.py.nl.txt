   241	                user_profile, current_energy, current_attention, task_data, cognitive_load

   266	    def _calculate_task_cognitive_load(
   267	        self,
   268	        complexity: float,
   269	        duration: int,
   270	        task_data: Dict[str, Any]
   271	    ) -> float:
   272	        """Calculate cognitive load for task."""
   273	        try:
   274	            # Base load from complexity
   275	            base_load = complexity * 0.4
   276	
   277	            # Duration factor (longer tasks = higher load)
   278	            duration_factor = min(duration / 60.0, 0.3)  # Max 0.3 for duration
   279	
   280	            # Task type factor
   281	            task_type_loads = {
   282	                "research": 0.2,
   283	                "implementation": 0.3,
   284	                "debugging": 0.4,
   285	                "architecture": 0.3,
   286	                "testing": 0.2,
   287	                "documentation": 0.1
   288	            }
   289	
   290	            task_description = task_data.get("description", "").lower()
   291	            task_type_load = 0.2  # Default
   292	
   293	            for task_type, load in task_type_loads.items():
   294	                if task_type in task_description:
   295	                    task_type_load = load
   296	                    break
   297	
   298	            # Dependencies factor (more dependencies = higher load)
   299	            dependencies = task_data.get("dependencies", [])
   300	            dependency_load = min(len(dependencies) * 0.05, 0.1)
   301	
   302	            # Total cognitive load
   303	            total_load = min(
   304	                base_load + duration_factor + task_type_load + dependency_load,
   305	                1.0
   306	            )
   307	
   308	            return total_load
   309	
   310	        except Exception as e:
   311	            return 0.5  # Default moderate load
   312	    def _assess_energy_match(
   313	        self,
   314	        current_energy: EnergyLevel,
   315	        cognitive_load: float,
   316	        profile: ADHDProfile
   317	    ) -> float:
   318	        """Assess how well task matches current energy level."""
   319	        try:
   320	            energy_scores = {
   321	                EnergyLevel.VERY_LOW: 0.1,
   322	                EnergyLevel.LOW: 0.3,
   323	                EnergyLevel.MEDIUM: 0.6,
   324	                EnergyLevel.HIGH: 0.8,
   325	                EnergyLevel.HYPERFOCUS: 1.0
   326	            }
   327	
   328	            current_capacity = energy_scores[current_energy]
   329	
   330	            # Perfect match gets highest score
   331	            if abs(current_capacity - cognitive_load) < 0.2:
   332	                return 1.0
   333	
   334	            # Penalize mismatches
   335	            mismatch_penalty = abs(current_capacity - cognitive_load)
   336	            energy_match = max(0.0, 1.0 - (mismatch_penalty * 2))
   337	
   338	            # Apply profile adjustments
   339	            if current_energy == EnergyLevel.HYPERFOCUS and profile.hyperfocus_tendency > 0.8:
   340	                # High hyperfocus tendency - can handle higher loads
   341	                energy_match += 0.2
   342	
   343	            return min(energy_match, 1.0)
   344	
   345	        except Exception as e:
   346	            return 0.5
   347	    def _assess_attention_compatibility(
   348	        self,
   349	        attention_state: AttentionState,
   350	        task_data: Dict[str, Any],
   351	        cognitive_load: float
   352	    ) -> float:
   353	        """Assess task compatibility with attention state."""
   354	        try:
   355	            compatibility_matrix = {
   356	                AttentionState.SCATTERED: {
   357	                    "max_cognitive_load": 0.3,
   358	                    "preferred_duration": 10,
   359	                    "complexity_penalty": 0.5
   360	                },
   361	                AttentionState.TRANSITIONING: {
   362	                    "max_cognitive_load": 0.4,
   363	                    "preferred_duration": 15,
   364	                    "complexity_penalty": 0.3
   365	                },
   366	                AttentionState.FOCUSED: {
   367	                    "max_cognitive_load": 0.8,
   368	                    "preferred_duration": 25,
   369	                    "complexity_penalty": 0.0
   370	                },
   371	                AttentionState.HYPERFOCUSED: {
   372	                    "max_cognitive_load": 1.0,
   373	                    "preferred_duration": 90,
   374	                    "complexity_penalty": -0.2  # Bonus for complex tasks
   375	                },
   376	                AttentionState.OVERWHELMED: {
   377	                    "max_cognitive_load": 0.1,
   378	                    "preferred_duration": 5,
   379	                    "complexity_penalty": 0.8
   380	                }
   381	            }
   382	
   383	            state_config = compatibility_matrix.get(attention_state, compatibility_matrix[AttentionState.FOCUSED])
   384	
   385	            # Check cognitive load compatibility
   386	            max_load = state_config["max_cognitive_load"]
   387	            if cognitive_load > max_load:
   388	                load_compatibility = max_load / max(cognitive_load, 0.1)
   389	            else:
   390	                load_compatibility = 1.0
   391	
   392	            # Check duration compatibility
   393	            task_duration = task_data.get("estimated_minutes", 25)
   394	            preferred_duration = state_config["preferred_duration"]
   395	            duration_compatibility = 1.0 - abs(task_duration - preferred_duration) / max(preferred_duration, task_duration)
   396	
   397	            # Apply complexity penalty/bonus
   398	            complexity_adjustment = state_config["complexity_penalty"]
   399	            adjusted_compatibility = max(0.0, min(1.0, (load_compatibility + duration_compatibility) / 2 - complexity_adjustment))
   400	
   401	            return adjusted_compatibility
   402	
   403	        except Exception as e:
   404	            return 0.5

   507	    async def _assess_current_energy_level(self, user_id: str) -> EnergyLevel:
   508	        """Assess current energy level for user."""
   509	        try:
   510	            # Get recent activity indicators
   511	            activity_data = await self._get_recent_activity(user_id)
   512	
   513	            # Analyze patterns
   514	            task_completion_rate = activity_data.get("completion_rate", 0.5)
   515	            context_switches = activity_data.get("context_switches", 0)
   516	            break_compliance = activity_data.get("break_compliance", 1.0)
   517	            time_since_last_break = activity_data.get("minutes_since_break", 0)
   518	
   519	            # Energy assessment algorithm
   520	            energy_score = 0.6  # Base energy level
   521	
   522	            # Task completion indicates energy
   523	            if task_completion_rate > 0.8:
   524	                energy_score += 0.3
   525	            elif task_completion_rate < 0.3:
   526	                energy_score -= 0.4
   527	
   528	            # High context switching indicates scattered energy
   529	            if context_switches > 5:
   530	                energy_score -= 0.3
   531	
   532	            # Break compliance indicates energy management
   533	            if break_compliance < 0.5:
   534	                energy_score -= 0.2
   535	
   536	            # Time since break affects energy
   537	            if time_since_last_break > 60:  # More than 1 hour
   538	                energy_score -= 0.3
   539	
   540	            # Map score to energy level
   541	            if energy_score >= 0.9:
   542	                current_energy = EnergyLevel.HYPERFOCUS
   543	            elif energy_score >= 0.7:
   544	                current_energy = EnergyLevel.HIGH
   545	            elif energy_score >= 0.4:
   546	                current_energy = EnergyLevel.MEDIUM
   547	            elif energy_score >= 0.2:
   548	                current_energy = EnergyLevel.LOW
   549	            else:
   550	                current_energy = EnergyLevel.VERY_LOW
   551	
   552	            # Update tracking
   553	            previous_energy = self.current_energy_levels.get(user_id)
   554	            self.current_energy_levels[user_id] = current_energy
   555	
   556	            # Log energy change if significant
   557	            if previous_energy and previous_energy != current_energy:
   558	                await self._log_energy_change(user_id, previous_energy, current_energy)
   559	
   560	            return current_energy
   561	
   562	        except Exception as e:
   563	            logger.error(f"Energy assessment failed for {user_id}: {e}")
   564	            return EnergyLevel.MEDIUM

   981	        """Calculate system-wide cognitive load."""
   982	        if not self.user_profiles:
