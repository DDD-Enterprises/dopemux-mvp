    46	class DopemuxEvent:
    47	    """Wrapper class to match test expectations (envelope/payload)."""
    48	    def __init__(self, envelope: Envelope, payload: Dict[str, Any], adhd_metadata: Optional[ADHDMetadata] = None):
    49	        self.envelope = envelope
    50	        self.payload = payload
    51	        self.adhd_metadata = adhd_metadata
    52	        self.source = None # Helpers
    53	        self.instance_id = None
    54	
    55	    @classmethod
    56	    def create(cls, event_type: str, namespace: str, payload: Dict[str, Any], 
    57	               priority: Priority = Priority.NORMAL, 
    58	               cognitive_load: Optional[CognitiveLoad] = None,
    59	               adhd_metadata: Optional[ADHDMetadata] = None,
    60	               source: Optional[str] = None,
    61	               instance_id: Optional[str] = None) -> 'DopemuxEvent':
    62	        envelope = Envelope(type=event_type, namespace=namespace, priority=priority)
    63	        instance = cls(envelope, payload, adhd_metadata)
    64	        instance.source = source
    65	        instance.instance_id = instance_id
    66	        return instance

    68	class EventBus(ABC):
    69	    @abstractmethod
    70	    async def publish(self, event: DopemuxEvent) -> bool:
    71	        ...
    72	
    73	    @abstractmethod
    74	    async def subscribe(self, pattern: str, callback: Callable[[DopemuxEvent], Any]) -> str:
    75	        """Subscribe to events matching pattern. Returns subscription ID."""
    76	        ...
    77	
    78	    @abstractmethod
    79	    async def unsubscribe(self, subscription_id: str) -> None:
    80	        ...

    82	class InMemoryAdapter(EventBus):
    83	    def __init__(self):
    84	        self._subscribers: Dict[str, Dict[str, Callable]] = {}
    85	        self._patterns: Dict[str, str] = {} # sub_id -> pattern
    86	
    87	    async def publish(self, event: DopemuxEvent) -> bool:
    88	        # Dispatch to all subscribers
    89	        for sub_id, pattern in self._patterns.items():
    90	            if self._matches(pattern, event):
    91	                callback = self._subscribers[pattern][sub_id]
    92	                try:
    93	                    if asyncio.iscoroutinefunction(callback):
    94	                        await callback(event)
    95	                    else:
    96	                        callback(event)
    97	                except Exception as e:
    98	                    logger.error(f"Error in subscriber {sub_id}: {e}")
    99	        return True
   100	
   101	    def _matches(self, pattern: str, event: Any) -> bool:
   102	        if hasattr(event, 'envelope') and hasattr(event.envelope, 'namespace'):
   103	             event_ns = event.envelope.namespace
   104	             if pattern.endswith('*'):
   105	                 if pattern == '*': return True
   106	                 prefix = pattern[:-1]
   107	                 return event_ns.startswith(prefix)
   108	             return event_ns == pattern
   109	        return True # Default match
   110	
   111	    async def subscribe(self, pattern: str, callback: Callable[[DopemuxEvent], Any]) -> str:
   112	        sub_id = str(uuid.uuid4())
   113	        if pattern not in self._subscribers:
   114	            self._subscribers[pattern] = {}
   115	        self._subscribers[pattern][sub_id] = callback
   116	        self._patterns[sub_id] = pattern
   117	        return sub_id
   118	
   119	    async def unsubscribe(self, subscription_id: str) -> None:
   120	        if subscription_id in self._patterns:
   121	            pattern = self._patterns[subscription_id]
   122	            if pattern in self._subscribers:
   123	                if subscription_id in self._subscribers[pattern]:
   124	                    del self._subscribers[pattern][subscription_id]
   125	            del self._patterns[subscription_id]

   127	class RedisStreamsAdapter(EventBus):
   128	    def __init__(self, redis_url: str):
   129	        self.redis_url = redis_url
   130	        self.connected = False
   131	        self._subscriptions: Dict[str, Dict[str, Any]] = {}
   132	    
   133	    async def connect(self):
   134	        self.connected = True
   135	        logger.info(f"Connected to Redis at {self.redis_url}")
   136	        
   137	    async def disconnect(self):
   138	        self.connected = False
   139	
   140	    async def publish(self, event: DopemuxEvent) -> bool:
   141	        if not self.connected:
   142	            logger.warning("RedisStreamsAdapter.publish called while disconnected; using local fan-out fallback")
   143	
   144	        for sub_id, subscription in list(self._subscriptions.items()):
   145	            pattern = subscription["pattern"]
   146	            callback = subscription["callback"]
   147	            if not self._matches(pattern, event):
   148	                continue
   149	            try:
   150	                if asyncio.iscoroutinefunction(callback):
   151	                    await callback(event)
   152	                else:
   153	                    callback(event)
   154	            except Exception as exc:
   155	                logger.error("Error in RedisStreamsAdapter subscriber %s: %s", sub_id, exc)
   156	        return True
   157	
   158	    async def subscribe(self, pattern: str, callback: Callable[[DopemuxEvent], Any]) -> str:
   159	        subscription_id = str(uuid.uuid4())
   160	        self._subscriptions[subscription_id] = {
   161	            "pattern": pattern,
   162	            "callback": callback,
   163	        }
   164	        return subscription_id
   165	
   166	    async def unsubscribe(self, subscription_id: str) -> None:
   167	        self._subscriptions.pop(subscription_id, None)
   168	
   169	    def _matches(self, pattern: str, event: DopemuxEvent) -> bool:
   170	        if pattern == "*":
   171	            return True
   172	        namespace = getattr(event.envelope, "namespace", "")
   173	        if pattern.endswith("*"):
   174	            return namespace.startswith(pattern[:-1])
   175	        return namespace == pattern
