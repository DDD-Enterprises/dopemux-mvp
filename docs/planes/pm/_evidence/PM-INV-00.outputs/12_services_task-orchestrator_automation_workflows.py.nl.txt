   255	    async def trigger_automation(
   256	        self,
   257	        trigger: AutomationTrigger,
   258	        trigger_data: Dict[str, Any]
   259	    ) -> None:
   260	        """Trigger automation workflows for specific event."""
   261	        try:
   262	            # Get handlers for this trigger
   263	            handlers = self.trigger_handlers.get(trigger, [])
   264	
   265	            if not handlers:
   266	                logger.debug(f"No automation handlers for trigger: {trigger.value}")
   267	                return
   268	
   269	            # Execute handlers with ADHD considerations
   270	            for handler in handlers:
   271	                # Check cognitive load before triggering
   272	                current_load = await self._get_current_cognitive_load()
   273	
   274	                if current_load > self.cognitive_load_threshold:
   275	                    # Defer automation to reduce cognitive burden
   276	                    logger.info(f"🧠 Deferring automation due to high cognitive load: {trigger.value}")
   277	                    await self._defer_automation(trigger, trigger_data, delay_minutes=5)
   278	                    continue
   279	
   280	                # Execute automation
   281	                try:
   282	                    await handler(trigger_data)
   283	                except Exception as e:
   284	                    logger.error(f"Automation handler failed for {trigger.value}: {e}")
   285	
   286	        except Exception as e:
   287	            logger.error(f"Automation trigger failed: {e}")

   289	    async def _execute_workflow(
   290	        self,
   291	        workflow: AutomationWorkflow,
   292	        trigger_data: Dict[str, Any]
   293	    ) -> bool:
   294	        """Execute specific automation workflow."""
   295	        try:
   296	            if not workflow.enabled:
   297	                return False
   298	
   299	            # Check workflow conditions
   300	            if not await self._check_workflow_conditions(workflow, trigger_data):
   301	                logger.debug(f"Workflow conditions not met: {workflow.id}")
   302	                return False
   303	
   304	            # Add to active automations
   305	            self.active_automations.add(workflow.id)
   306	
   307	            logger.info(f"🤖 Executing automation: {workflow.automation_type.value}")
   308	
   309	            # Execute workflow actions in priority order
   310	            actions = sorted(workflow.actions, key=lambda a: a.get("priority", 5))
   311	
   312	            for action in actions:
   313	                success = await self._execute_workflow_action(
   314	                    action, workflow, trigger_data
   315	                )
   316	
   317	                if not success:
   318	                    logger.warning(f"Workflow action failed: {action['type']}")
   319	                    # Continue with other actions rather than failing entire workflow
   320	
   321	            # Update workflow execution metadata
   322	            workflow.last_executed = datetime.now(timezone.utc)
   323	            workflow.execution_count += 1
   324	            workflow.success_rate = (
   325	                ((workflow.success_rate * (workflow.execution_count - 1)) + 1.0)
   326	                / workflow.execution_count
   327	            )
   328	
   329	            # Remove from active automations
   330	            self.active_automations.discard(workflow.id)
   331	
   332	            self.automation_stats["workflows_executed"] += 1
   333	            logger.info(f"✅ Automation completed: {workflow.automation_type.value}")
   334	
   335	            return True
   336	
   337	        except Exception as e:
   338	            logger.error(f"Workflow execution failed: {e}")
   339	            workflow.execution_count += 1
   340	            workflow.success_rate = (
   341	                (workflow.success_rate * (workflow.execution_count - 1))
   342	                / workflow.execution_count
   343	            )
   344	            self.active_automations.discard(workflow.id)
   345	            return False

   406	                # Check for ADHD risk factors

   412	            sprint_analysis = {
   413	                "sprint_id": sprint_id,
   414	                "total_tasks": len(sprint_tasks),

   892	        title = str(task.get("title", task.get("name", ""))).lower()
   893	        description = str(task.get("description", "")).lower()

   899	        score = 0.15 + min(0.6, overlap * 0.15)

   906	        if task_id and task_id in normalized_path:
