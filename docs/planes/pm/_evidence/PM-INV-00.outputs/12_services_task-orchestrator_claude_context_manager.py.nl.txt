   311	    async def _context_monitor(self) -> None:
   312	        """Background monitor for context changes."""
   313	        logger.info("👁️ Started Claude.md context monitoring")
   314	
   315	        while self.running:
   316	            try:
   317	                # Check for context updates needed
   318	                context_changed = await self._check_context_changes()
   319	
   320	                if context_changed or self.pending_updates:
   321	                    await self._update_claude_context()
   322	
   323	                # Monitor every 2 minutes
   324	                await asyncio.sleep(120)
   325	
   326	            except Exception as e:
   327	                logger.error(f"Context monitoring error: {e}")
   328	                await asyncio.sleep(300)

   333	        self,
   334	        sprint_id: str,
   335	        sprint_name: str,
   336	        mode: ContextMode,
   337	        task_data: List[Dict[str, Any]] = None
   338	    ) -> None:
   339	        """Update context for new sprint."""
   340	        try:
   341	            # Update context
   342	            self.current_context.active_sprint_id = sprint_id
   343	            self.current_context.sprint_name = sprint_name
   344	            self.current_context.sprint_mode = mode
   345	
   346	            if task_data:
   347	                # Extract priority tasks (high priority, not blocked)
   348	                priority_tasks = [
   349	                    task for task in task_data
   350	                    if task.get("priority", 5) <= 3 and task.get("status") != "blocked"
   351	                ]
   352	                self.current_context.priority_tasks = priority_tasks[:5]  # Top 5 for ADHD
   353	
   354	            # Mark for update
   355	            self.pending_updates.add("sprint_context")
   356	
   357	            logger.info(f"🎯 Sprint context updated: {sprint_name} ({mode.value})")

   452	    async def _generate_dynamic_claude_md(self) -> str:
   453	        """Generate dynamic Claude.md content based on current context."""
   454	        try:
   455	            # Start with base content
   456	            if not self.base_claude_md:
   457	                content = self._get_default_claude_md()
   458	            else:
   459	                content = self.base_claude_md
   460	
   461	            # Find dynamic context insertion point
   462	            context_marker = "# Dynamic Context - Auto-Generated"
   463	            context_end_marker = "# End Dynamic Context"
   464	
   465	            # Remove existing dynamic context
   466	            if context_marker in content:
   467	                start_pos = content.find(context_marker)
   468	                end_pos = content.find(context_end_marker)
   469	
   470	                if end_pos > start_pos:
   471	                    end_pos += len(context_end_marker)
   472	                    content = content[:start_pos] + content[end_pos:]
   473	
   474	            # Generate new dynamic context
   475	            dynamic_context = await self._generate_context_section()
   476	
   477	            # Insert at the beginning for visibility
   478	            insertion_point = content.find("\n## ")  # After main title
   479	            if insertion_point > 0:
   480	                content = (
   481	                    content[:insertion_point] + "\n\n" +
   482	                    dynamic_context + "\n" +
   483	                    content[insertion_point:]
   484	                )
   485	            else:
   486	                content = dynamic_context + "\n\n" + content
   487	
   488	            return content
   489	
   490	        except Exception as e:
   491	            logger.error(f"Dynamic Claude.md generation failed: {e}")
   492	            return self.base_claude_md or self._get_default_claude_md()

   507	                context.adhd_state,

   511	            # Template variables
   512	            template_vars = {
   513	                "sprint_name": context.sprint_name or "No active sprint",
   514	                "sprint_id": context.active_sprint_id or "none",
   515	                "active_task_title": context.active_task_title or "No active task",
   516	                "current_focus_area": context.current_focus_area or "General development",
   517	                "task_count": len(context.priority_tasks),
   518	                "complex_tasks": len([t for t in context.priority_tasks if t.get("complexity_score", 0) > 0.7]),
   519	                "adhd_state": context.adhd_state.value,
   520	                "energy_level": context.energy_level,
   521	                "break_frequency": context.break_frequency,
   522	                "recommended_task_duration": context.recommended_task_duration,
   523	                "cognitive_load_status": self._get_cognitive_load_status(),
   524	                "focus_mode_status": self._get_focus_mode_status(),
   525	                "automation_success_rate": self._get_automation_success_rate(),

   553	    def _render_template(self, template: str, variables: Dict[str, Any]) -> str:
   554	        """Render template with variables."""
   555	        try:
   556	            rendered = template
   557	
   558	            # Simple variable substitution
   559	            for key, value in variables.items():
   560	                placeholder = f"{{{{{key}}}}}"
   561	                rendered = rendered.replace(placeholder, str(value))
   562	
   563	            # Handle conditional blocks
   564	            rendered = self._process_conditional_blocks(rendered, variables)
   565	
   566	            return rendered
   567	
   568	        except Exception as e:
   569	            logger.error(f"Template rendering failed: {e}")
   570	            return template

   572	    def _process_conditional_blocks(self, content: str, variables: Dict[str, Any]) -> str:
   573	        """Process conditional template blocks."""
   574	        try:
   575	            # Handle {{#each array}} blocks
   576	            each_pattern = r'\{\{#each (\w+)\}\}(.*?)\{\{/each\}\}'
   577	
   578	            def replace_each(match):
   579	                array_name = match.group(1)
   580	                block_content = match.group(2)
   581	                array_data = variables.get(array_name, [])
   582	
   583	                if not array_data:
   584	                    return ""
   585	
   586	                result = []
   587	                for item in array_data:
   588	                    item_content = block_content
   589	                    # Replace item properties
   590	                    if isinstance(item, dict):
   591	                        for key, value in item.items():
   592	                            item_content = item_content.replace(f"{{{{{key}}}}}", str(value))
   593	                    result.append(item_content)
   594	
   595	                return "".join(result)

   605	        """Apply ADHD-specific adaptations to content."""

   609	            # Limit instruction count
   610	            max_instructions = adhd_config.get("max_instructions", 10)

   615	                # Keep most important instructions

   617	                adapted = '\n'.join([
   618	                    line for line in lines
   619	                    if not line.strip().startswith('-') or line in important_lines

   624	            if instruction_length == "short":
