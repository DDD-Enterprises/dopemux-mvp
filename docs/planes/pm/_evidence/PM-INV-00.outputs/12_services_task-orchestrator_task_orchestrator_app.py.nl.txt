    44	@asynccontextmanager
    45	async def lifespan(app: FastAPI):
    46	    """Application lifespan handler for startup/shutdown."""
    47	    # Startup
    48	    logger.info("=" * 60)
    49	    logger.info("🚀 Task-Orchestrator v2.0 - HTTP Mode")
    50	    logger.info("=" * 60)
    51	    logger.info(f"📊 Instance: {settings.instance_name}")
    52	    logger.info(f"🔌 Port: {settings.port}")
    53	    logger.info("=" * 60)
    54	    
    55	    try:
    56	        # Initialize connections
    57	        await leantime_client.initialize()
    58	        await redis_manager.initialize()
    59	        
    60	        # Initialize components
    61	        agent_coordinator.initialize_pool()
    62	        await adhd_monitor.start_monitoring()

    64	        logger.info("✅ Task-Orchestrator initialized successfully")
    65	    except Exception as e:
    66	        logger.error(f"❌ Startup failed: {e}")
    67	        raise
    68	    
    69	    yield
    70	    
    71	    # Shutdown
    72	    logger.info("🛑 Shutting down Task-Orchestrator...")
    73	    await adhd_monitor.stop_monitoring()
    74	    await leantime_client.close()
    75	    await redis_manager.close()

   113	@app.post("/api/tools/{tool_name}")
   114	async def call_tool(tool_name: str, request: Request):
   115	    """Execute an MCP tool."""
   116	    try:
   117	        arguments = await request.json()
   118	        logger.info(f"🔧 Executing tool: {tool_name}")
   119	        
   120	        result = await handle_tool_call(tool_name, arguments)
   121	        
   122	        if "error" in result:
   123	            logger.warning(f"❌ Tool execution failed: {result['error']}")
   124	            # We return 200 even on logical error to match MCP protocol behavior style,
   125	            # but usually REST implies 4xx/5xx. DopeconBridge expects 200 with result
   126	            # or it raises HTTP exception. 
   127	            # If result has "error", DopeconBridge might treat it as result.
   128	            return result
   129	            
   130	        return result
   131	        
   132	    except Exception as e:
   133	        logger.error(f"❌ Tool call error: {e}")
   134	        raise HTTPException(status_code=500, detail=str(e))
   135

   137	@app.post("/api/decompose", response_model=DecompositionResponse)
   138	async def decompose_task(request: DecompositionRequest):
   139	    """
   140	    Decompose a complex task into ADHD-friendly micro-tasks.
   141	    
   142	    Called by ADHD Engine when automatic decomposition is triggered.
   143	    
   144	    Flow:
   145	    1. Get task from internal storage
   146	    2. Call Pal planner for AI decomposition
   147	    3. Convert to OrchestrationTask objects
   148	    4. Persist to ConPort (parent BLOCKED, subtasks TODO, DECOMPOSED_INTO links)
   149	    5. Sync to Leantime (create child tickets for team visibility)
   150	    6. Return structured breakdown
   151	    """
   152	    try:
   153	        logger.info(f"📋 Decomposition request for task {request.task_id}")
   154	        
   155	        # Get singleton instances
   156	        from task_orchestrator.core import (
   157	            get_task_coordinator,
   158	            get_pal_client,
   159	            get_conport_adapter
   160	        )
   161	        
   162	        task_coordinator = get_task_coordinator()
   163	        pal_client = get_pal_client()
   164	        conport_adapter = get_conport_adapter()
   165	        
   166	        # Execute decomposition
   167	        response = await handle_decomposition_request(
   168	            request=request,
   169	            task_coordinator=task_coordinator,
   170	            pal_client=pal_client,
   171	            leantime_client=leantime_client,
   172	            conport_adapter=conport_adapter,
   173	            workspace_id=settings.workspace_id
   174	        )
   175	        
   176	        logger.info(
   177	            f"✅ Decomposition complete: {len(response.subtask_ids)} subtasks, "
   178	            f"{response.total_estimated_minutes}min total"
   179	        )
   180	        
   181	        return response
   182	    
   183	    except ValueError as e:
   184	        logger.error(f"❌ Decomposition validation error: {e}")
   185	        raise HTTPException(status_code=404, detail=str(e))
   186	    except Exception as e:
   187	        logger.error(f"❌ Decomposition failed: {e}")
   188	        raise HTTPException(status_code=500, detail=str(e))
   189
