     1	"""
     2	ADHD-Aware Task Prioritization - Standalone Demo
     3	
     4	Demonstrates intelligent task prioritization based on:
     5	- Energy level matching (high/medium/low)
     6	- Complexity + attention state matching
     7	- ADHD suitability scoring
     8	
     9	This is a design/proof-of-concept that shows how task recommendations
    10	should work when integrated into Task-Orchestrator.
    11	
    12	Note: enhanced_orchestrator.py has structural issues (Component 5 methods
    13	are after `if __name__`). This standalone demo shows the correct logic.
    14	"""
    15	
    16	import asyncio
    17	
    18	import logging
    19	
    20	logger = logging.getLogger(__name__)
    21	
    22	from dataclasses import dataclass
    23	from enum import Enum
    24	from typing import List, Dict, Any
    25	from datetime import datetime
    26	
    27	
    28	class EnergyLevel(str, Enum):
    29	    """Energy levels for task matching."""
    30	    HIGH = "high"
    31	    MEDIUM = "medium"
    32	    LOW = "low"
    33	
    34	
    35	class AttentionState(str, Enum):
    36	    """Attention states for ADHD accommodation."""
    37	    FOCUSED = "focused"
    38	    SCATTERED = "scattered"
    39	    HYPERFOCUS = "hyperfocus"
    40	
    41	
    42	@dataclass
    43	class UserState:
    44	    """Current user state for ADHD optimization."""
    45	    energy: EnergyLevel
    46	    attention: AttentionState
    47	    session_duration_minutes: int
    48	
    49	
    50	class ADHDTaskPrioritizer:
    51	    """
    52	    ADHD-aware task prioritization engine.
    53	
    54	    Scores and ranks tasks based on energy/complexity/attention matching.
    55	    """
    56	
    57	    def calculate_adhd_match_score(
    58	        self,
    59	        task_complexity: float,
    60	        task_energy: str,
    61	        user_state: UserState
    62	    ) -> float:
    63	        """
    64	        Calculate how well a task matches user's current ADHD state.
    65	
    66	        Args:
    67	            task_complexity: 0.0-1.0 (higher = more complex)
    68	            task_energy: "high", "medium", or "low"
    69	            user_state: Current user state
    70	
    71	        Returns:
    72	            Score 0.0-1.0 (higher = better match)
    73	        """
    74	        score = 0.5  # Base score
    75	
    76	        # Energy match (most important +/-0.3)
    77	        if task_energy == user_state.energy.value:
    78	            score += 0.3  # Perfect energy match
    79	        elif task_energy == "low" and user_state.energy.value == "medium":
    80	            score += 0.2  # Low tasks okay for medium energy
    81	        elif task_energy == "medium" and user_state.energy.value == "high":
    82	            score += 0.15  # Medium tasks okay for high energy
    83	        else:
    84	            score -= 0.2  # Energy mismatch penalty
    85	
    86	        # Complexity + attention match (+/-0.3)
    87	        if user_state.attention == AttentionState.FOCUSED:
    88	            # Focused: prefer optimal complexity (0.4-0.7)
    89	            if 0.4 <= task_complexity <= 0.7:
    90	                score += 0.2
    91	            elif task_complexity < 0.4:
    92	                score += 0.1  # Simple tasks okay
    93	            else:
    94	                score -= 0.1  # Too complex for sustained focus
    95	
    96	        elif user_state.attention == AttentionState.SCATTERED:
    97	            # Scattered: prefer simple tasks (<0.3)
    98	            if task_complexity < 0.3:
    99	                score += 0.3  # Perfect for scattered state
   100	            elif task_complexity < 0.5:
   101	                score += 0.1  # Moderate okay
   102	            else:
   103	                score -= 0.3  # Too complex when scattered
   104	
   105	        elif user_state.attention == AttentionState.HYPERFOCUS:
   106	            # Hyperfocus: can handle complex (>0.6)
   107	            if task_complexity > 0.6:
   108	                score += 0.2  # Good use of hyperfocus
   109	            elif task_complexity > 0.4:
   110	                score += 0.1  # Okay
   111	            else:
   112	                score -= 0.1  # Waste of hyperfocus state
   113	
   114	        # Clamp to 0.0-1.0
   115	        return max(0.0, min(1.0, score))
   116	
   117	    def generate_recommendation_reason(
   118	        self,
   119	        score: float,
   120	        task_energy: str,
   121	        task_complexity: float,
   122	        user_state: UserState
   123	    ) -> str:
   124	        """Generate human-readable reason for recommendation."""
   125	
   126	        if score >= 0.8:
   127	            return f"Perfect match: {task_energy} energy + {task_complexity:.1f} complexity suits your {user_state.attention.value} state"
   128	        elif score >= 0.6:
   129	            return f"Good match: {task_energy} energy aligns with current state"
   130	        elif score >= 0.4:
   131	            return f"Acceptable: Doable in your current {user_state.attention.value} state"
   132	        else:
   133	            return f"Consider later: Better when energy is {task_energy}"
   134	
   135	    def prioritize_tasks(
   136	        self,
   137	        tasks: List[Dict[str, Any]],
   138	        user_state: UserState,
   139	        limit: int = 5
   140	    ) -> List[Dict[str, Any]]:
   141	        """
   142	        Prioritize tasks based on ADHD match.
   143	
   144	        Args:
   145	            tasks: List of tasks with complexity_score and energy_required
   146	            user_state: Current user state
   147	            limit: Max recommendations to return
   148	
   149	        Returns:
   150	            Sorted list of task recommendations
   151	        """
   152	        # Score all tasks
   153	        scored_tasks = []
   154	        for task in tasks:
   155	            complexity = task.get("complexity_score", 0.5)
   156	            energy_required = task.get("energy_required", "medium")
   157	
   158	            score = self.calculate_adhd_match_score(
   159	                task_complexity=complexity,
   160	                task_energy=energy_required,
   161	                user_state=user_state
   162	            )
   163	
   164	            scored_tasks.append({
   165	                "task": task,
   166	                "score": score,
   167	                "complexity": complexity,
   168	                "energy_required": energy_required
   169	            })
   170	
   171	        # Sort by score (highest first)
   172	        scored_tasks.sort(key=lambda x: x["score"], reverse=True)
   173	
   174	        # Build recommendations
   175	        recommendations = []
   176	        for i, scored in enumerate(scored_tasks[:limit]):
   177	            task = scored["task"]
   178	            score = scored["score"]
   179	
   180	            reason = self.generate_recommendation_reason(
   181	                score=score,
   182	                task_energy=scored["energy_required"],
   183	                task_complexity=scored["complexity"],
   184	                user_state=user_state
   185	            )
   186	
   187	            recommendations.append({
   188	                "task_id": task["task_id"],
   189	                "title": task["title"],
   190	                "reason": reason,
   191	                "confidence": score,
   192	                "priority": i + 1,
   193	                "adhd_match": {
   194	                    "energy_match": scored["energy_required"] == user_state.energy.value,
   195	                    "complexity": scored["complexity"],
   196	                    "user_energy": user_state.energy.value,
   197	                    "user_attention": user_state.attention.value
   198	                }
   199	            })
   200	
   201	        return recommendations
   202
