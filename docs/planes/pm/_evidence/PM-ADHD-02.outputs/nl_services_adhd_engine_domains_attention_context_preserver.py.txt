    18	@dataclass
    19	class PreservedContext:
    20	    """Snapshot of development context at a moment in time."""
    21	    # Code context
    22	    active_files: List[str] = field(default_factory=list)
    23	    cursor_positions: Dict[str, int] = field(default_factory=dict)  # file -> line number
    24	    recent_symbols: List[str] = field(default_factory=list)  # Recently accessed functions/classes
    25	    
    26	    # Mental model
    27	    mental_model_summary: str = ""  # AI-generated summary of what user was working on
    28	    task_description: Optional[str] = None  # Current task from Leantime/ConPort
    29	    
    30	    # Temporal context
    31	    timestamp: datetime = field(default_factory=datetime.now)
    32	    session_duration: int = 0  # Minutes in current session
    33	    
    34	    # Decision trail
    35	    recent_decisions: List[Dict[str, Any]] = field(default_factory=list)  # Last 3 decisions from ConPort
    36	    
    37	    # Workspace
    38	    workspace_path: Optional[str] = None
    39	    git_branch: Optional[str] = None
    40	    
    41	    def to_dict(self) -> Dict[str, Any]:
    42	        """Convert to JSON-serializable dict."""
    43	        data = asdict(self)
    44	        data['timestamp'] = self.timestamp.isoformat()
    45	        return data
    46	    
    47	    @classmethod
    48	    def from_dict(cls, data: Dict[str, Any]) -> 'PreservedContext':
    49	        """Restore from dict."""
    50	        data['timestamp'] = datetime.fromisoformat(data['timestamp'])
    51	        return cls(**data)

   146	    async def restore_context(
   147	        self,
   148	        user_id: str,
   149	        context: Optional[PreservedContext] = None
   150	    ) -> Dict[str, Any]:
   151	        """
   152	        Restore context after break.
   153	        
   154	        Args:
   155	            user_id: User identifier
   156	            context: Optional specific context to restore (default: most recent)
   157	        
   158	        Returns:
   159	            Summary for user display
   160	        """
   161	        logger.info(f"🔄 Restoring context for {user_id}")
   162	        
   163	        try:
   164	            # Load context if not provided
   165	            if context is None:
   166	                context = await self._load_latest_snapshot(user_id)
   167	            
   168	            if context is None:
   169	                return {"error": "No saved context found"}
   170	            
   171	            # Build restoration summary
   172	            time_since = (datetime.now() - context.timestamp).total_seconds() / 60
   173	            
   174	            summary = {
   175	                "time_since_save": f"{int(time_since)} minutes ago",
   176	                "task": context.task_description or "Unknown task",
   177	                "mental_model": context.mental_model_summary,
   178	                "active_files": context.active_files,
   179	                "git_branch": context.git_branch,
   180	                "recent_decisions": [d.get('summary', '') for d in context.recent_decisions],
   181	                "restoration_tips": self._generate_restoration_tips(context)
   182	            }
   183	            
   184	            logger.info(f"✅ Context restored from {int(time_since)} minutes ago")
   185	            return summary
   186	            
   187	        except Exception as e:
   188	            logger.error(f"Failed to restore context: {e}")
   189	            return {"error": str(e)}

   216	    async def _get_git_branch(self, workspace_path: str) -> Optional[str]:
   217	        """Get current git branch."""
   218	        try:
   219	            import subprocess
   220	            result = subprocess.run(
   221	                ['git', 'branch', '--show-current'],
   222	                cwd=workspace_path,
   223	                capture_output=True,
   224	                text=True,
   225	                timeout=2
   226	            )
   227	            if result.returncode == 0:
   228	                return result.stdout.strip()
   229	        except Exception as e:
   230	            logger.warning(f"Failed to get git branch: {e}")
   231	        return None

   233	    async def _generate_mental_model(self, context: PreservedContext) -> str:
   234	        """
   235	        Generate AI summary of what user was working on.
   236	        
   237	        Uses recent decisions + active files to create human-readable summary.
   238	        """
   239	        # Simple heuristic implementation
   240	        parts = []
   241	        
   242	        if context.task_description:
   243	            parts.append(f"Working on: {context.task_description}")
   244	        
   245	        if context.active_files:
   246	            parts.append(f"Editing {len(context.active_files)} files: {', '.join(context.active_files[:3])}")
   247	        
   248	        if context.git_branch:
   249	            parts.append(f"On branch: {context.git_branch}")
   250	        
   251	        if not parts:
   252	            return "No clear task context captured"
   253	        
   254	        return " | ".join(parts)

   256	    async def _save_snapshot(self, user_id: str, context: PreservedContext) -> bool:
   257	        """Save context snapshot to disk."""
   258	        try:
   259	            filename = f"{self.storage_path}/{user_id}_latest.json"
   260	            with open(filename, 'w') as f:
   261	                json.dump(context.to_dict(), f, indent=2)
   262	            return True
   263	        except Exception as e:
   264	            logger.error(f"Failed to save snapshot: {e}")
   265	            return False
   266	    
   267	    async def _load_latest_snapshot(self, user_id: str) -> Optional[PreservedContext]:
   268	        """Load most recent context snapshot."""
   269	        try:
   270	            filename = f"{self.storage_path}/{user_id}_latest.json"
   271	            with open(filename, 'r') as f:
   272	                data = json.load(f)
   273	            return PreservedContext.from_dict(data)
   274	        except FileNotFoundError:
   275	            logger.warning(f"No saved context for {user_id}")
   276	            return None
   277	        except Exception as e:
   278	            logger.error(f"Failed to load snapshot: {e}")
   279	            return None
