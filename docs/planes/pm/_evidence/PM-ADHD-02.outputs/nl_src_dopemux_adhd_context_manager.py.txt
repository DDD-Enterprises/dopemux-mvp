   116	    def _validate_project_path(self, file_path: Path) -> bool:
   117	        """
   118	        Ensure file path is within project boundaries to prevent directory traversal.
   119	
   120	        Args:
   121	            file_path: Path to validate
   122	
   123	        Returns:
   124	            True if path is safe and within project, False otherwise
   125	        """
   126	        try:
   127	            # Resolve both paths to handle symlinks and relative paths
   128	            resolved_file = file_path.resolve()
   129	            resolved_project = self.project_path.resolve()
   130	
   131	            # Check if file path is within project boundaries
   132	            resolved_file.relative_to(resolved_project)
   133	            return True
   134	
   135	        except (ValueError, OSError):
   136	            # ValueError: path is outside project
   137	            # OSError: path doesn't exist or permission issues
   138	            console.log(f"[red]Security: Blocked access to path outside project: {file_path}[/red]")
   139	            return False

   141	    def _run_git_command(self, args: List[str], timeout: int = 10) -> Optional[str]:
   142	        """
   143	        Securely run git command with input validation and timeout.
   144	
   145	        Args:
   146	            args: Git command arguments (without 'git')
   147	            timeout: Command timeout in seconds
   148	
   149	        Returns:
   150	            Command output if successful, None if failed
   151	        """
   152	        # Validate git command arguments
   153	        allowed_commands = {
   154	            "branch", "status", "log", "show", "diff", "rev-parse", "config"
   155	        }
   156	
   157	        if not args or args[0] not in allowed_commands:
   158	            console.log(f"[red]Security: Git command not allowed: {args}[/red]")
   159	            return None
   160	
   161	        # Build secure command
   162	        cmd = ["git"] + args
   163	
   164	        try:
   165	            # Run with security constraints
   166	            result = subprocess.run(
   167	                cmd,
   168	                cwd=str(self.project_path),  # Ensure we're in project directory
   169	                capture_output=True,
   170	                text=True,
   171	                timeout=timeout,
   172	                check=False  # Don't raise on non-zero exit
   173	            )
   174	
   175	            if result.returncode == 0:
   176	                return result.stdout.strip()
   177	            else:
   178	                # Log error but don't expose it to prevent information leakage
   179	                console.log(f"[yellow]Git command failed (code {result.returncode})[/yellow]")
   180	                return None
   181	
   182	        except subprocess.TimeoutExpired:
   183	            console.log(f"[red]Git command timed out after {timeout}s[/red]")
   184	            return None
   185	        except Exception as e:
   186	            console.log(f"[red]Git command error: {type(e).__name__}[/red]")
   187	            return None

   197	                    working_directory TEXT NOT NULL,

   199	                    hash TEXT NOT NULL,
   200	                    size INTEGER NOT NULL,
   201	                    created_at TEXT DEFAULT CURRENT_TIMESTAMP

   203	            """
   204	            )
   205	
   206	            conn.execute(
   207	                """
   208	                CREATE TABLE IF NOT EXISTS session_metadata (
   209	                    session_id TEXT PRIMARY KEY,
   210	                    project_path TEXT NOT NULL,
   211	                    started_at TEXT NOT NULL,
   212	                    last_active TEXT NOT NULL,
   213	                    total_duration INTEGER DEFAULT 0,

   243	    def save_context(self, message: Optional[str] = None, force: bool = False) -> str:
   244	        """
   245	        Save current development context.
   246	
   247	        Args:
   248	            message: Optional message describing the save
   249	            force: Force save even if no changes detected
   250	
   251	        Returns:
   252	            Session ID of saved context
   253	        """
   254	        try:
   255	            # Capture current state
   256	            context = self._capture_current_state()
   257	            if message:
   258	                context.message = message
   259	
   260	            # Check if context has changed (unless forced)
   261	            if not force and not self._has_context_changed(context):
   262	                return context.session_id
   263	
   264	            # Store in database
   265	            self._store_context(context)
   266	
   267	            # Update session metadata
   268	            self._update_session_metadata(context.session_id)
   269	
   270	            # Save detailed session file
   271	            self._save_session_file(context)
   272	
   273	            self._current_session_id = context.session_id
   274	            return context.session_id
   275	
   276	        except Exception as e:
   277	            console.log(f"[red]Error saving context: {e}[/red]")
   278	            # Emergency fallback
   279	            emergency_session_id = self._emergency_save()
   280	            return emergency_session_id or str(uuid.uuid4())

   438	    def generate_smart_description(self, context: ContextSnapshot) -> str:
   439	        """Generate smart description based on git changes and context."""
   440	        git_state = context.git_state
   441	
   442	        # Check git status for recent changes
   443	        if git_state.get("status"):
   444	            status_lines = git_state["status"].split("\n")
   445	            modified_files = []
   446	            added_files = []
   447	            deleted_files = []
   448	
   449	            for line in status_lines:
   450	                line = line.strip()
   451	                if line.startswith("M "):
   452	                    modified_files.append(line[2:])
   453	                elif line.startswith("A "):
   454	                    added_files.append(line[2:])
   455	                elif line.startswith("D "):
   456	                    deleted_files.append(line[2:])
   457	
   458	            # Generate description based on changes
   459	            if added_files:
   460	                if len(added_files) == 1:
   461	                    return f"Added {Path(added_files[0]).name}"
   462	                else:
   463	                    return f"Added {len(added_files)} new files"
   464	            elif modified_files:
   465	                # Determine primary file type
   466	                extensions = [Path(f).suffix for f in modified_files]
   467	                common_ext = max(set(extensions), key=extensions.count) if extensions else ""
   468	
   469	                if len(modified_files) == 1:
   470	                    return f"Modified {Path(modified_files[0]).name}"
   471	                elif common_ext == ".py":
   472	                    return f"Updated Python code ({len(modified_files)} files)"
   473	                elif common_ext in [".js", ".ts", ".tsx"]:
   474	                    return f"Updated JavaScript/TypeScript ({len(modified_files)} files)"
   475	                elif common_ext == ".md":
   476	                    return f"Updated documentation ({len(modified_files)} files)"
   477	                else:
   478	                    return f"Modified {len(modified_files)} files"
   479	            elif deleted_files:
   480	                return f"Removed {len(deleted_files)} files"
   481	
   482	        # Fallback to goal-based description
   483	        goal = context.current_goal
   484	        if goal and goal != "Continue development":
   485	            return goal[:50] + ("..." if len(goal) > 50 else "")
   486	
   487	        # Check last commit for context
   488	        last_commit = git_state.get("last_commit", "")
   489	        if last_commit and len(last_commit.split()) > 1:
   490	            # Extract meaningful part of commit message
   491	            commit_msg = " ".join(last_commit.split()[1:])
   492	            return f"Continue: {commit_msg[:40]}..."
   493	
   494	        return "Development session"

   496	    def auto_tag_session(self, context: ContextSnapshot) -> List[str]:
   497	        """Automatically generate tags based on session context."""
   498	        tags = []
   499	
   500	        # Analyze git changes for automatic tagging
   501	        git_state = context.git_state
   502	        if git_state.get("status"):
   503	            status = git_state["status"].lower()
   504	
   505	            # File type tags
   506	            if ".py" in status:
   507	                tags.append("python")
   508	            if any(ext in status for ext in [".js", ".ts", ".tsx", ".jsx"]):
   509	                tags.append("javascript")
   510	            if ".md" in status or "readme" in status:
   511	                tags.append("docs")
   512	            if "test" in status or ".test." in status:
   513	                tags.append("testing")
   514	            if any(ext in status for ext in [".yml", ".yaml", ".json", "config"]):
   515	                tags.append("config")
   516	
   517	        # Analyze file paths for context
   518	        open_files = context.open_files
   519	        file_paths = []
   520	        for file_info in open_files:
   521	            if isinstance(file_info, dict):
   522	                path = file_info.get("path", "")
   523	            else:
   524	                path = str(file_info)
   525	            file_paths.append(path.lower())
   526	
   527	        combined_paths = " ".join(file_paths)
   528	
   529	        # Feature detection
   530	        if any(keyword in combined_paths for keyword in ["feature", "feat", "new"]):
   531	            tags.append("feature")
   532	        if any(keyword in combined_paths for keyword in ["fix", "bug", "error"]):
   533	            tags.append("bugfix")
   534	        if any(keyword in combined_paths for keyword in ["test", "spec", "__test__"]):
   535	            tags.append("testing")
   536	        if any(keyword in combined_paths for keyword in ["doc", "readme", "guide"]):
   537	            tags.append("documentation")
   538	        if any(keyword in combined_paths for keyword in ["refactor", "clean"]):
   539	            tags.append("refactor")
   540	
   541	        # Analyze message for manual tags
   542	        message = context.message.lower() if context.message else ""
   543	        if message:
   544	            if any(keyword in message for keyword in ["wip", "progress", "working"]):
   545	                tags.append("wip")
   546	            if any(keyword in message for keyword in ["done", "complete", "finish"]):
   547	                tags.append("complete")
   548	            if any(keyword in message for keyword in ["break", "pause", "save"]):
   549	                tags.append("checkpoint")
   550	
   551	        # Time-based tags
   552	        hour = datetime.now().hour
   553	        if 6 <= hour < 12:
   554	            tags.append("morning")
   555	        elif 12 <= hour < 17:
   556	            tags.append("afternoon")
   557	        elif 17 <= hour < 22:
   558	            tags.append("evening")
   559	        else:
   560	            tags.append("late-night")
   561	
   562	        # Focus duration tags
   563	        focus_duration = context.focus_duration
   564	        if focus_duration > 45:
   565	            tags.append("deep-work")
   566	        elif focus_duration > 20:
   567	            tags.append("focused")
   568	        elif focus_duration > 0:
   569	            tags.append("quick-session")
   570	
   571	        return list(set(tags))  # Remove duplicates

   573	    def detect_session_type(self, context: ContextSnapshot) -> str:
   574	        """Detect session type based on context and changes."""
   575	        message = context.message.lower() if context.message else ""
   576	        git_state = context.git_state
   577	
   578	        # Check commit message patterns
   579	        if git_state.get("last_commit"):
   580	            last_commit = git_state["last_commit"].lower()
   581	            if any(word in last_commit for word in ["feat:", "feature:", "add:"]):
   582	                return "feature"
   583	            elif any(word in last_commit for word in ["fix:", "bug:", "hotfix:"]):
   584	                return "bugfix"
   585	            elif any(word in last_commit for word in ["docs:", "doc:"]):
   586	                return "documentation"
   587	            elif any(word in last_commit for word in ["test:", "tests:"]):
   588	                return "testing"
   589	            elif any(word in last_commit for word in ["refactor:", "clean:"]):
   590	                return "refactor"
   591	            elif any(word in last_commit for word in ["config:", "setup:"]):
   592	                return "configuration"
   593	
   594	        # Check message for type indicators
   595	        if any(keyword in message for keyword in ["implement", "add", "create", "new"]):
   596	            return "feature"
   597	        elif any(keyword in message for keyword in ["fix", "bug", "error", "issue"]):
   598	            return "bugfix"
   599	        elif any(keyword in message for keyword in ["doc", "readme", "guide", "explain"]):
   600	            return "documentation"
   601	        elif any(keyword in message for keyword in ["test", "coverage", "spec"]):
   602	            return "testing"
   603	        elif any(keyword in message for keyword in ["refactor", "clean", "optimize"]):
   604	            return "refactor"
   605	        elif any(keyword in message for keyword in ["debug", "investigate", "explore"]):
   606	            return "debugging"
   607	        elif any(keyword in message for keyword in ["review", "check", "validate"]):
   608	            return "review"
   609	
   610	        # Check files being worked on
   611	        open_files = context.open_files
   612	        test_files = 0
   613	        doc_files = 0
   614	        config_files = 0

   616	        for file_info in open_files:

   631	        total_files = len(open_files)

   636	                return "documentation"

   829	    def _store_context(self, context: ContextSnapshot) -> None:
   830	        """Store context in database."""
   831	        data_json = json.dumps(context.to_dict())
   832	        data_hash = hashlib.sha256(data_json.encode()).hexdigest()
   833	
   834	        with sqlite3.connect(self.db_path) as conn:
   835	            conn.execute(
   836	                """INSERT OR REPLACE INTO context_snapshots
   837	                   (session_id, timestamp, working_directory, data, hash, size)
   838	                   VALUES (?, ?, ?, ?, ?, ?)""",
   839	                (
   840	                    context.session_id,
   841	                    context.timestamp,
   842	                    context.working_directory,
   843	                    data_json,
   844	                    data_hash,
   845	                    len(data_json),
   846	                ),
   847	            )
   848	
   849	            # Store tags in separate table
   850	            if context.tags:
   851	                # Clear existing tags for this session
   852	                conn.execute(
   853	                    "DELETE FROM session_tags WHERE session_id = ?",
   854	                    (context.session_id,)
   855	                )
   856	                # Insert new tags
   857	                for tag in context.tags:
   858	                    conn.execute(
   859	                        """INSERT INTO session_tags (session_id, tag)
   860	                           VALUES (?, ?)""",
   861	                        (context.session_id, tag)
   862	                    )
