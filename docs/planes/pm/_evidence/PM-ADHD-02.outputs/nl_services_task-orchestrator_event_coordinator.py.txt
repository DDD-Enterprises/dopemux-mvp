   208	    async def _start_event_workers(self) -> None:
   209	        """Start event processing workers with priority handling."""
   210	        # Start workers for each priority level
   211	        priority_workers = {
   212	            EventPriority.CRITICAL: 3,   # 3 workers for critical events
   213	            EventPriority.HIGH: 2,       # 2 workers for high priority
   214	            EventPriority.MEDIUM: 2,     # 2 workers for medium priority
   215	            EventPriority.LOW: 1,        # 1 worker for low priority
   216	            EventPriority.BACKGROUND: 1  # 1 worker for background
   217	        }
   218	
   219	        for priority, worker_count in priority_workers.items():
   220	            for i in range(worker_count):
   221	                worker = asyncio.create_task(
   222	                    self._event_worker(priority, f"{priority.name.lower()}-{i}")
   223	                )
   224	                self.workers.append(worker)
   225	
   226	        # Start telemetry logger
   227	        telemetry_worker = asyncio.create_task(self._telemetry_logger())
   228	        self.workers.append(telemetry_worker)
   229	
   230	        logger.info(f"👥 Started {len(self.workers)} event processing workers (including telemetry logger)")

   232	    async def _event_worker(self, priority: EventPriority, worker_id: str) -> None:
   233	        """Event processing worker for specific priority level."""
   234	        logger.debug(f"🚀 Event worker {worker_id} started")
   235	
   236	        while self.running:
   237	            try:
   238	                # Get event from priority queue
   239	                queue = self.event_queues[priority]
   240	                _, event = await asyncio.wait_for(queue.get(), timeout=5.0)
   241	
   242	                # Process event with ADHD considerations
   243	                await self._process_event(event, worker_id)
   244	
   245	            except asyncio.TimeoutError:
   246	                # No events at this priority, continue
   247	                continue
   248	            except Exception as e:
   249	                logger.error(f"Event worker {worker_id} error: {e}")
   250	                await asyncio.sleep(1.0)
   251	
   252	        logger.debug(f"🛑 Event worker {worker_id} stopped")

   282	    async def _process_event(self, event: CoordinationEvent, worker_id: str) -> None:
   283	        """Process individual coordination event."""
   284	        try:
   285	            start_time = datetime.now(timezone.utc)
   286	
   287	            # Apply ADHD filtering
   288	            if not await self._should_process_event(event):
   289	                self.metrics["events_filtered"] += 1
   290	                logger.debug(f"🎯 Event filtered (ADHD): {event.event_type}")
   291	                return
   292	
   293	            # Check if event has expired
   294	            if event.expires_at and datetime.now(timezone.utc) > event.expires_at:
   295	                logger.warning(f"⏰ Event expired: {event.id}")
   296	                self._record_suppression(event, "expiry")
   297	                return
   298	
   299	            # Get processors for this event type
   300	            processors = self.event_processors.get(event.event_type, [])
   301	
   302	            if not processors:
   303	                logger.warning(f"No processors registered for {event.event_type}")
   304	                return
   305	
   306	            # Execute processors in parallel
   307	            results = await asyncio.gather(
   308	                *[processor(event) for processor in processors],
   309	                return_exceptions=True
   310	            )
   311	
   312	            # Check for errors
   313	            errors = [r for r in results if isinstance(r, Exception)]
   314	            if errors:
   315	                logger.error(f"Event processing errors: {errors}")
   316	                event.error_message = str(errors[0])
   317	
   318	            # Update processing metadata
   319	            event.processed_at = datetime.now(timezone.utc)
   320	            processing_time = (event.processed_at - start_time).total_seconds()
   321	
   322	            # Store processing history in Redis
   323	            await self._store_event_history(event, processing_time)
   324	
   325	            self.metrics["events_processed"] += 1
   326	            logger.debug(f"✅ Event processed by {worker_id}: {event.event_type} in {processing_time:.3f}s")
   327	
   328	        except Exception as e:
   329	            logger.error(f"Event processing failed: {e}")
   330	            event.attempts += 1
   331	            event.error_message = str(e)
   332	
   333	            # Retry logic with exponential backoff
   334	            if event.attempts < event.max_attempts:
   335	                delay = 2 ** event.attempts  # Exponential backoff
   336	                logger.info(f"🔄 Retrying event {event.id} in {delay}s (attempt {event.attempts + 1})")
   337	
   338	                # Re-queue with delay
   339	                asyncio.create_task(self._delayed_requeue(event, delay))

   341	    async def _delayed_requeue(self, event: CoordinationEvent, delay: float) -> None:
   342	        """Re-queue event after delay."""
   343	        await asyncio.sleep(delay)
   344	        await self.emit_event(event)

   348	    async def emit_event(self, event: CoordinationEvent) -> bool:
   349	        """Emit coordination event for processing."""

   385	    async def _should_process_event(self, event: CoordinationEvent) -> bool:
   386	        """Determine if event should be processed based on ADHD state."""
   387	        try:
   388	            # Track that we received an event
   389	            self.suppression_telemetry.events_received += 1
   390	            event_type_str = event.event_type.value
   391	            priority_str = event.priority.name
   392	
   393	            # Initialize per-event-type tracking
   394	            if event_type_str not in self.suppression_telemetry.per_event_type:
   395	                self.suppression_telemetry.per_event_type[event_type_str] = {"received": 0, "suppressed": 0}
   396	            self.suppression_telemetry.per_event_type[event_type_str]["received"] += 1
   397	
   398	            # Initialize per-priority tracking
   399	            if priority_str not in self.suppression_telemetry.per_priority:
   400	                self.suppression_telemetry.per_priority[priority_str] = {"received": 0, "suppressed": 0}
   401	            self.suppression_telemetry.per_priority[priority_str]["received"] += 1
   402	
   403	            # Apply custom filters
   404	            for filter_func in self.event_filters:
   405	                if not await filter_func(event):
   406	                    self._record_suppression(event, "custom_filter")
   407	                    return False
   408	
   409	            # Check focus mode restrictions
   410	            if self.current_focus_mode == "deep":
   411	                # In deep focus, only allow critical events
   412	                if event.priority not in [EventPriority.CRITICAL, EventPriority.HIGH]:
   413	                    self._record_suppression(event, "deep_focus_priority")
   414	                    return False
   415	
   416	                # Block interrupting events unless critical
   417	                if not event.interruption_allowed and event.priority != EventPriority.CRITICAL:
   418	                    self._record_suppression(event, "deep_focus_interrupt")
   419	                    return False
   420	
   421	            # Check energy level matching
   422	            if self.current_energy_level == "low":
   423	                # Low energy: avoid high cognitive load events
   424	                if event.cognitive_load > 0.7:
   425	                    self._record_suppression(event, "energy_level")
   426	                    return False
   427	
   428	            # Check for event flooding (ADHD protection)
   429	            recent_events = await self._get_recent_event_count(event.event_type)
   430	            if recent_events > 10:  # More than 10 similar events in last minute
   431	                logger.warning(f"🌊 Event flooding detected: {event.event_type}")
   432	                self._record_suppression(event, "flood")
   433	                return False
   434	
   435	            # Event passed all filters
   436	            self.suppression_telemetry.events_passed += 1
   437	            return True
   438	
   439	        except Exception as e:
   440	            logger.error(f"Event filtering failed: {e}")
   441	            return True  # Default to processing on error

   453	    def _record_suppression(self, event: CoordinationEvent, rule: str) -> None:
   454	        """Record event suppression with rule and metadata."""
   455	        # Increment overall suppression counter
   456	        self.suppression_telemetry.events_suppressed += 1
   457	
   458	        # Increment per-rule counter
   459	        rule_attr = f"suppressed_by_{rule}"
   460	        current_count = getattr(self.suppression_telemetry, rule_attr, 0)
   461	        setattr(self.suppression_telemetry, rule_attr, current_count + 1)
   462	
   463	        # Update per-event-type suppression
   464	        event_type_str = event.event_type.value
   465	        if event_type_str in self.suppression_telemetry.per_event_type:
   466	            self.suppression_telemetry.per_event_type[event_type_str]["suppressed"] += 1
   467	
   468	        # Update per-priority suppression
   469	        priority_str = event.priority.name
   470	        if priority_str in self.suppression_telemetry.per_priority:
   471	            self.suppression_telemetry.per_priority[priority_str]["suppressed"] += 1

   897	        suppression_rate_pct = (

   909	                "events_suppressed": t.events_suppressed,
   910	                "signal_noise_ratio": round(signal_noise_ratio, 3),
   911	                "suppression_rate_pct": round(suppression_rate_pct, 1),
   912	                "runtime_minutes": round(runtime_minutes, 1)
