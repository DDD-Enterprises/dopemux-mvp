     2	ADHD Workflow Manager for Dopemux
     3	
     4	Provides ADHD-optimized workflow management with:
     5	- Cognitive load monitoring and automatic adjustments
     6	- 25-minute focus session tracking with break suggestions
     7	- Progressive disclosure of information
     8	- Context preservation across interruptions
     9	- Integration with ADHD Engine services
    10	
    11	Core Features:
    12	- Real-time cognitive load assessment
    13	- Automatic break recommendations
    14	- Session state management
    15	- Visual feedback optimization
    16	- Interrupt recovery support
    17	"""
    18	
    19	import asyncio
    20	import time
    21	from typing import Dict, Any, Optional, List, Tuple
    22	from datetime import datetime, timedelta

    28	logger = logging.getLogger(__name__)

    31	class ADHDWorkflowManager:
    32	    """
    33	    Manages ADHD-optimized development workflows.
    34	
    35	    Coordinates between ADHD Engine services, user interactions,
    36	    and Dopemux operations to maintain optimal cognitive flow.
    37	    """
    38	
    39	    def __init__(self):
    40	        self.session_start_time = None
    41	        self.last_break_time = None
    42	        self.cognitive_load_history = []
    43	        self.current_focus_level = "medium"
    44	        self.session_duration_minutes = 25  # ADHD-optimized focus period
    45	        self.break_duration_minutes = 5
    46	
    47	        # Initialize UI components
    48	        self.progress_display = ProgressDisplay()
    49	        self.interactive_prompts = InteractivePrompts()
    50	
    51	        # ADHD Engine integration (will be initialized if available)
    52	        self.adhd_engine_client = None
    53	
    54	    def start_session(self, task_description: str = "") -> Dict[str, Any]:
    55	        """
    56	        Start an ADHD-optimized work session.
    57	
    58	        Args:
    59	            task_description: Description of the current task
    60	
    61	        Returns:
    62	            Session initialization data
    63	        """
    64	        self.session_start_time = datetime.now()
    65	        self.cognitive_load_history = []
    66	        self.current_focus_level = "medium"
    67	
    68	        session_data = {
    69	            "session_id": f"session_{int(time.time())}",
    70	            "start_time": self.session_start_time.isoformat(),
    71	            "task_description": task_description,
    72	            "focus_level": self.current_focus_level,
    73	            "target_duration": self.session_duration_minutes,
    74	            "break_duration": self.break_duration_minutes
    75	        }
    76	
    77	        logger.info(f"Started ADHD session: {session_data['session_id']}")
    78	        return session_data
    79	
    80	    def end_session(self) -> Dict[str, Any]:
    81	        """
    82	        End the current work session.
    83	
    84	        Returns:
    85	            Session summary data
    86	        """
    87	        if not self.session_start_time:
    88	            return {"error": "No active session"}
    89	
    90	        end_time = datetime.now()
    91	        duration = (end_time - self.session_start_time).total_seconds() / 60
    92	
    93	        session_summary = {
    94	            "session_id": f"session_{int(self.session_start_time.timestamp())}",
    95	            "start_time": self.session_start_time.isoformat(),
    96	            "end_time": end_time.isoformat(),
    97	            "duration_minutes": round(duration, 1),
    98	            "cognitive_load_avg": self._calculate_avg_cognitive_load(),
    99	            "breaks_taken": 1 if self.last_break_time else 0,
   100	            "focus_level_final": self.current_focus_level
   101	        }
   102	
   103	        # Reset session state
   104	        self.session_start_time = None
   105	        self.last_break_time = None
   106	
   107	        logger.info(f"Ended ADHD session: {session_summary['duration_minutes']}min")
   108	        return session_summary
   109	
   110	    def check_break_needed(self) -> Tuple[bool, str]:
   111	        """
   112	        Check if a break is needed based on session timing and cognitive load.
   113	
   114	        Returns:
   115	            Tuple of (break_needed, reason)
   116	        """
   117	        if not self.session_start_time:
   118	            return False, "No active session"
   119	
   120	        elapsed_minutes = (datetime.now() - self.session_start_time).total_seconds() / 60
   121	
   122	        # Time-based break check
   123	        if elapsed_minutes >= self.session_duration_minutes:
   124	            return True, f"Session duration reached ({elapsed_minutes:.1f}min)"
   125	
   126	        # Cognitive load check
   127	        avg_load = self._calculate_avg_cognitive_load()
   128	        if avg_load > 0.8:
   129	            return True, f"High cognitive load ({avg_load:.1f})"
   130	
   131	        # Recent break check (prevent too frequent breaks)
   132	        if self.last_break_time:
   133	            time_since_break = (datetime.now() - self.last_break_time).total_seconds() / 60
   134	            if time_since_break < 20:  # Minimum 20min between breaks
   135	                return False, "Recent break taken"
   136	
   137	        return False, "No break needed"
   138	
   139	    def suggest_break(self) -> Dict[str, Any]:
   140	        """
   141	        Generate a break suggestion with timing and activities.
   142	
   143	        Returns:
   144	            Break suggestion data
   145	        """
   146	        break_needed, reason = self.check_break_needed()
   147	
   148	        if not break_needed:
   149	            return {
   150	                "suggested": False,
   151	                "reason": reason
   152	            }
   153	
   154	        # Generate break activities based on cognitive load
   155	        activities = self._generate_break_activities()
   156	
   157	        return {
   158	            "suggested": True,
   159	            "reason": reason,
   160	            "duration_minutes": self.break_duration_minutes,
   161	            "activities": activities,
   162	            "urgency": "high" if reason.startswith("High cognitive") else "medium"
   163	        }
   164	
   165	    def update_cognitive_load(self, load_level: float, context: str = "") -> None:
   166	        """
   167	        Update the current cognitive load level.
   168	
   169	        Args:
   170	            load_level: Float 0.0-1.0 (0=low, 1=high cognitive load)
   171	            context: Optional context about the load measurement
   172	        """
   173	        timestamp = datetime.now()
   174	
   175	        self.cognitive_load_history.append({
   176	            "timestamp": timestamp,
   177	            "load": load_level,
   178	            "context": context
   179	        })
   180	
   181	        # Update focus level based on load
   182	        if load_level < 0.3:
   183	            self.current_focus_level = "low"
   184	        elif load_level < 0.7:
   185	            self.current_focus_level = "medium"
   186	        else:
   187	            self.current_focus_level = "high"
   188	
   189	        # Keep only last 50 measurements (about 25 minutes at 30s intervals)
   190	        if len(self.cognitive_load_history) > 50:
   191	            self.cognitive_load_history = self.cognitive_load_history[-50:]
   192	
   193	        logger.debug(f"Cognitive load updated: {load_level:.2f} ({self.current_focus_level})")
   194	
   195	    def take_break(self) -> Dict[str, Any]:
   196	        """
   197	        Record that a break was taken.
   198	
   199	        Returns:
   200	            Break completion data
   201	        """
   202	        self.last_break_time = datetime.now()
   203	
   204	        return {
   205	            "break_taken": True,
   206	            "timestamp": self.last_break_time.isoformat(),
   207	            "duration_minutes": self.break_duration_minutes,
   208	            "next_session_start": (self.last_break_time + timedelta(minutes=self.break_duration_minutes)).isoformat()
   209	        }
   210	
   211	    def get_progressive_info(self, full_info: Dict[str, Any], user_expertise: str = "intermediate") -> Dict[str, Any]:
   212	        """
   213	        Apply progressive disclosure to information based on user context.
   214	
   215	        Args:
   216	            full_info: Complete information dictionary
   217	            user_expertise: User expertise level ("beginner", "intermediate", "expert")
   218	
   219	        Returns:
   220	            Filtered information appropriate for user level
   221	        """
   222	        if user_expertise == "beginner":
   223	            # Show minimal essential info
   224	            essential_keys = ["status", "message", "next_action"]
   225	            return {k: v for k, v in full_info.items() if k in essential_keys}
   226	
   227	        elif user_expertise == "intermediate":
   228	            # Show moderate detail, hide advanced options
   229	            hide_keys = ["debug_info", "internal_metrics", "raw_data"]
   230	            return {k: v for k, v in full_info.items() if k not in hide_keys}
   231	
   232	        else:  # expert
   233	            # Show everything
   234	            return full_info
   235	
   236	    def get_context_snapshot(self) -> Dict[str, Any]:
   237	        """
   238	        Get a complete context snapshot for interrupt recovery.
   239	
   240	        Returns:
   241	            Current session state for restoration
   242	        """
   243	        return {
   244	            "session_active": self.session_start_time is not None,
   245	            "session_start": self.session_start_time.isoformat() if self.session_start_time else None,
   246	            "last_break": self.last_break_time.isoformat() if self.last_break_time else None,
   247	            "current_focus_level": self.current_focus_level,
   248	            "cognitive_load_recent": self._calculate_avg_cognitive_load(),
   249	            "session_duration_minutes": self.session_duration_minutes,
   250	            "cognitive_load_history_count": len(self.cognitive_load_history)
   251	        }
   252	
   253	    def restore_context(self, snapshot: Dict[str, Any]) -> bool:
   254	        """
   255	        Restore session context from a snapshot.
   256	
   257	        Args:
   258	            snapshot: Context snapshot from get_context_snapshot()
   259	
   260	        Returns:
   261	            Success status
   262	        """
   263	        try:
   264	            if snapshot.get("session_active"):
   265	                self.session_start_time = datetime.fromisoformat(snapshot["session_start"])
   266	                if snapshot.get("last_break"):
   267	                    self.last_break_time = datetime.fromisoformat(snapshot["last_break"])
   268	                self.current_focus_level = snapshot.get("current_focus_level", "medium")
   269	
   270	            logger.info("Context restored successfully")
   271	            return True
   272	        except Exception as e:
   273	            logger.error(f"Context restoration failed: {e}")
   274	            return False
   275	
   276	    def _calculate_avg_cognitive_load(self, window_minutes: int = 10) -> float:
   277	        """Calculate average cognitive load over recent window."""
   278	        if not self.cognitive_load_history:
   279	            return 0.5  # Default medium load
   280	
   281	        # Get measurements within time window
   282	        cutoff_time = datetime.now() - timedelta(minutes=window_minutes)
   283	        recent_loads = [
   284	            entry["load"] for entry in self.cognitive_load_history
   285	            if entry["timestamp"] > cutoff_time
   286	        ]
   287	
   288	        return sum(recent_loads) / len(recent_loads) if recent_loads else 0.5
   289	
   290	    def _generate_break_activities(self) -> List[str]:
   291	        """Generate appropriate break activities based on cognitive load."""
   292	        avg_load = self._calculate_avg_cognitive_load()
   293	
   294	        if avg_load > 0.8:
   295	            # High cognitive load - suggest calming activities
   296	            return [
   297	                "Take a 2-minute breathing exercise",
   298	                "Step away from screen and look at something 20+ feet away",
   299	                "Do gentle neck and shoulder stretches",
   300	                "Drink water and have a healthy snack"
   301	            ]
   302	        elif avg_load > 0.6:
   303	            # Medium cognitive load - suggest light activities
   304	            return [
   305	                "Stand up and walk around for 2 minutes",
   306	                "Do some simple desk stretches",
   307	                "Listen to 1 minute of calming music",
   308	                "Practice the 4-7-8 breathing technique"
   309	            ]
   310	        else:
   311	            # Lower cognitive load - suggest engagement activities
   312	            return [
   313	                "Take a short walk outside if possible",
   314	                "Do some light physical activity",
   315	                "Listen to an interesting podcast segment",
   316	                "Have a conversation with a colleague"
   317	            ]
