     1	"""
     2	Event type definitions for Dopemux event bus.
     3	
     4	All events are Pydantic models for type safety and validation.
     5	Based on UI Design Research Synthesis (Decision #25).
     6	"""
     7	
     8	from datetime import datetime
     9	from enum import Enum
    10	from typing import Any, Dict, Optional
    11	from pydantic import BaseModel, Field
    12	
    13	
    14	class EventPriority(str, Enum):
    15	    """Event priority levels for ADHD-optimized filtering."""
    16	
    17	    LOW = "low"           # Background updates, non-critical
    18	    MEDIUM = "medium"     # Normal updates
    19	    HIGH = "high"         # Important state changes
    20	    CRITICAL = "critical" # Requires immediate attention
    21	
    22	
    23	class Event(BaseModel):
    24	    """Base event class for all Dopemux events."""
    25	
    26	    type: str = Field(..., description="Event type identifier")
    27	    timestamp: datetime = Field(default_factory=datetime.now)
    28	    priority: EventPriority = Field(default=EventPriority.MEDIUM)
    29	    data: Dict[str, Any] = Field(default_factory=dict)
    30	    source: Optional[str] = Field(default=None, description="Component that emitted event")
    31	
    32	    class Config:
    33	        frozen = True  # Events are immutable
    34	
    35	
    36	class WorktreeEvent(Event):
    37	    """Events related to git worktree operations."""
    38	
    39	    type: str = "worktree"
    40	
    41	    class Action(str, Enum):
    42	        CREATED = "created"
    43	        SWITCHED = "switched"
    44	        REMOVED = "removed"
    45	        CLEANED = "cleaned"
    46	
    47	    action: Action = Field(..., description="Worktree action performed")
    48	    branch: str = Field(..., description="Branch name")
    49	    path: Optional[str] = Field(default=None, description="Worktree path")
    50	
    51	
    52	class ContextEvent(Event):
    53	    """Events related to ConPort context changes."""
    54	
    55	    type: str = "context"
    56	
    57	    class Action(str, Enum):
    58	        UPDATED = "updated"
    59	        RESTORED = "restored"
    60	        DECISION_LOGGED = "decision_logged"
    61	        PROGRESS_UPDATED = "progress_updated"
    62	
    63	    action: Action = Field(..., description="Context action performed")
    64	    workspace_id: str = Field(..., description="ConPort workspace ID")
    65	    context_type: str = Field(..., description="product_context or active_context")
    66	
    67	
    68	class ADHDEvent(Event):
    69	    """Events related to ADHD state changes from ADHD Engine."""
    70	
    71	    type: str = "adhd"
    72	
    73	    class StateType(str, Enum):
    74	        ENERGY = "energy"
    75	        ATTENTION = "attention"
    76	        BREAK_REMINDER = "break_reminder"
    77	        CONTEXT_SWITCH = "context_switch"
    78	
    79	    state_type: StateType = Field(..., description="Type of ADHD state")
    80	    previous_value: Optional[str] = Field(default=None)
    81	    current_value: str = Field(..., description="New state value")
    82	    confidence: float = Field(default=1.0, ge=0.0, le=1.0)
    83	
    84	
    85	class ThemeEvent(Event):
    86	    """Events related to theme changes."""
    87	
    88	    type: str = "theme"
    89	
    90	    class Action(str, Enum):
    91	        SWITCHED = "switched"
    92	        UPDATED = "updated"
    93	        INTERPOLATED = "interpolated"  # Energy state transition
    94	
    95	    action: Action = Field(..., description="Theme action")
    96	    theme_name: str = Field(..., description="Theme identifier")
    97	    previous_theme: Optional[str] = Field(default=None)
    98	
    99	
   100	class SessionEvent(Event):
   101	    """Events related to tmux session lifecycle."""
   102	
   103	    type: str = "session"
   104	
   105	    class Action(str, Enum):
   106	        CREATED = "created"
   107	        ATTACHED = "attached"
   108	        DETACHED = "detached"
   109	        DESTROYED = "destroyed"
   110	        PANE_CREATED = "pane_created"
   111	        LAYOUT_CHANGED = "layout_changed"
   112	
   113	    action: Action = Field(..., description="Session action")
   114	    session_id: str = Field(..., description="Tmux session ID")
   115	    session_name: Optional[str] = Field(default=None)
