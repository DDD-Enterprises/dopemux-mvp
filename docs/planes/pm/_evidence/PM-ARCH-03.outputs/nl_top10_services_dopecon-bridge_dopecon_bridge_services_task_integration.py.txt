    32	    async def parse_prd_to_tasks(self, prd_content: str, project_id: str) -> List[Task]:
    33	        """
    34	        Parse PRD using Task-Master-AI and create shared task structure.
    35	        ADHD-friendly: Clear progress indicators and manageable chunks.
    36	        """
    37	        logger.info(f"🔍 Parsing PRD for project {project_id} (instance: {settings.instance_name})")
    38	
    39	        try:
    40	            # Step 1: Use Task-Master-AI to parse PRD
    41	            prd_result = await self.mcp_manager.call_tool(
    42	                "task-master-ai",
    43	                "parse_prd",
    44	                {"content": prd_content, "project_id": project_id}
    45	            )
    46	
    47	            # Step 2: Convert to unified task format and store in shared database
    48	            tasks = []
    49	            async with await self.db_manager.get_session() as session:
    50	                for task_data in prd_result.get("tasks", []):
    51	                    task = Task(
    52	                        id=str(uuid.uuid4()),
    53	                        title=task_data.get("title", ""),
    54	                        description=task_data.get("description", ""),
    55	                        status=TaskStatus.PLANNED,
    56	                        priority=TaskPriority(task_data.get("priority", "medium")),
    57	                        project_id=project_id,
    58	                        instance_id=settings.instance_name,
    59	                        tags=task_data.get("tags", [])
    60	                    )
    61	
    62	                    # Store in shared database
    63	                    task_record = TaskRecord(**asdict(task))
    64	                    session.add(task_record)
    65	                    tasks.append(task)
    66	
    67	                await session.commit()
    68	
    69	            # Step 3: Use Task-Orchestrator to analyze dependencies
    70	            await self._analyze_task_dependencies(tasks)
    71	
    72	            # Step 4: Create tasks in Leantime for tracking
    73	            await self._sync_tasks_to_leantime(tasks)
    74	
    75	            # Step 5: Cache results for fast access
    76	            cache_client = await self.cache_manager.get_client()
    77	            await cache_client.setex(
    78	                f"project:{project_id}:tasks",
    79	                3600,  # 1 hour cache
    80	                json.dumps([asdict(task) for task in tasks], default=str)
    81	            )
    82	
    83	            logger.info(f"✅ Successfully processed {len(tasks)} tasks from PRD")
    84	            return tasks
    85	
    86	        except Exception as e:
    87	            logger.error(f"❌ PRD parsing failed: {e}")
    88	            raise

    90	    async def _analyze_task_dependencies(self, tasks: List[Task]):
    91	        """Use Task-Orchestrator to analyze and set dependencies."""
    92	        try:
    93	            task_descriptions = [
    94	                {"id": t.id, "title": t.title, "description": t.description}
    95	                for t in tasks
    96	            ]
    97	
    98	            dependency_result = await self.mcp_manager.call_tool(
    99	                "task-orchestrator",
   100	                "analyze_dependencies",
   101	                {"tasks": task_descriptions}
   102	            )
   103	
   104	            # Update dependencies in shared database
   105	            async with await self.db_manager.get_session() as session:
   106	                for dep in dependency_result.get("dependencies", []):
   107	                    task_id = dep.get("task_id")
   108	                    depends_on = dep.get("depends_on", [])
   109	
   110	                    await session.execute(
   111	                        TaskRecord.__table__.update()
   112	                        .where(TaskRecord.id == task_id)
   113	                        .values(dependencies=depends_on, updated_at=datetime.utcnow())
   114	                    )
   115	
   116	                    for task in tasks:
   117	                        if task.id == task_id:
   118	                            task.dependencies = depends_on
   119	
   120	                await session.commit()
   121	
   122	        except Exception as e:
   123	            logger.warning(f"⚠️ Dependency analysis failed: {e}")

   125	    async def _sync_tasks_to_leantime(self, tasks: List[Task]):
   126	        """Sync tasks to Leantime for project management tracking."""
   127	        try:
   128	            for task in tasks:
   129	                leantime_result = await self.mcp_manager.call_tool(
   130	                    "leantime-bridge",
   131	                    "create_ticket",
   132	                    {
   133	                        "projectId": int(task.project_id) if task.project_id else 1,
   134	                        "headline": task.title,
   135	                        "description": task.description,
   136	                        "priority": self._map_priority_to_leantime(task.priority),
   137	                        "type": "task"
   138	                    }
   139	                )
   140	
   141	                if "id" in leantime_result:
   142	                    task.tags.append(f"leantime_id:{leantime_result['id']}")
   143	
   144	                    async with await self.db_manager.get_session() as session:
   145	                        await session.execute(
   146	                            TaskRecord.__table__.update()
   147	                            .where(TaskRecord.id == task.id)
   148	                            .values(tags=task.tags, updated_at=datetime.utcnow())
   149	                        )
   150	                        await session.commit()
   151	
   152	        except Exception as e:
   153	            logger.warning(f"⚠️ Leantime sync failed: {e}")

   155	    def _map_priority_to_leantime(self, priority: TaskPriority) -> str:
   156	        """Map unified priority to Leantime priority format."""
   157	        mapping = {
   158	            TaskPriority.LOW: "1",
   159	            TaskPriority.MEDIUM: "2",
   160	            TaskPriority.HIGH: "3",
   161	            TaskPriority.CRITICAL: "4"
   162	        }
   163	        return mapping.get(priority, "2")

   165	    async def get_next_actionable_tasks(self, project_id: str, limit: int = 5) -> List[Task]:
   166	        """
   167	        Get next actionable tasks for ADHD-friendly workflow.
   168	        Checks shared database for cross-instance task coordination.
   169	        """
   170	        try:
   171	            cache_client = await self.cache_manager.get_client()
   172	            cache_key = f"project:{project_id}:actionable_tasks"
   173	            cached_result = await cache_client.get(cache_key)
   174	
   175	            if cached_result:
   176	                logger.debug("📋 Returning cached actionable tasks")
   177	                task_dicts = json.loads(cached_result)
   178	                return [
   179	                    Task(
   180	                        id=t["id"],
   181	                        title=t["title"],
   182	                        description=t["description"],
   183	                        status=TaskStatus(t["status"]),
   184	                        priority=TaskPriority(t["priority"]),
   185	                        instance_id=t.get("instance_id", "default"),
   186	                        project_id=t.get("project_id"),
   187	                        dependencies=t.get("dependencies", []),
   188	                        tags=t.get("tags", [])
   189	                    )
   190	                    for t in task_dicts
   191	                ]
   192	
   193	            # Query database for actionable tasks (no blocking dependencies)
   194	            async with await self.db_manager.get_session() as session:
   195	                from sqlalchemy import text
   196	                
   197	                result = await session.execute(
   198	                    text("""
   199	                    SELECT t.* FROM tasks t
   200	                    WHERE t.project_id = :project_id
   201	                    AND t.status = 'planned'
   202	                    AND (
   203	                        t.dependencies IS NULL
   204	                        OR t.dependencies = '[]'::json
   205	                    )
   206	                    ORDER BY
   207	                        CASE t.priority
   208	                            WHEN 'critical' THEN 1
   209	                            WHEN 'high' THEN 2
   210	                            WHEN 'medium' THEN 3
   211	                            WHEN 'low' THEN 4
   212	                        END,
   213	                        t.created_at
   214	                    LIMIT :limit
   215	                    """),
   216	                    {"project_id": project_id, "limit": limit}
   217	                )
   218	
   219	                task_records = result.fetchall()
   220	                actionable_tasks = [
   221	                    Task(
   222	                        id=record.id,
   223	                        title=record.title,
   224	                        description=record.description or "",
   225	                        status=TaskStatus(record.status),
   226	                        priority=TaskPriority(record.priority),
   227	                        instance_id=record.instance_id,
   228	                        project_id=record.project_id,
   229	                        dependencies=record.dependencies or [],
   230	                        tags=record.tags or [],
   231	                        assigned_to=record.assigned_to,
   232	                        estimated_hours=record.estimated_hours
   233	                    )
   234	                    for record in task_records
   235	                ]
   236	
   237	                # Cache result for 5 minutes
   238	                await cache_client.setex(
   239	                    cache_key,
   240	                    300,
   241	                    json.dumps([asdict(task) for task in actionable_tasks], default=str)
   242	                )
   243	
   244	                logger.info(f"📋 Found {len(actionable_tasks)} actionable tasks for project {project_id}")
   245	                return actionable_tasks
   246	
   247	        except Exception as e:
   248	            logger.error(f"❌ Failed to get actionable tasks: {e}")
   249	            return []

   251	    async def update_task_status(
   252	        self,
   253	        task_id: str,
   254	        new_status: TaskStatus,
   255	        assigned_to: str = None
   256	    ) -> Dict[str, Any]:
   257	        """
   258	        Update task status across all systems with dependency resolution.
   259	        ADHD-friendly: Clear progress feedback and automatic next-action suggestions.
   260	        """
   261	        logger.info(f"🔄 Updating task {task_id} to status {new_status.value}")
   262	
   263	        try:
   264	            async with await self.db_manager.get_session() as session:
   265	                result = await session.execute(
   266	                    TaskRecord.__table__.select().where(TaskRecord.id == task_id)
   267	                )
   268	                task_record = result.fetchone()
   269	
   270	                if not task_record:
   271	                    raise ValueError(f"Task {task_id} not found")
   272	
   273	                old_status = task_record.status
   274	
   275	                update_data = {
   276	                    "status": new_status.value,
   277	                    "updated_at": datetime.utcnow()
   278	                }
   279	                if assigned_to:
   280	                    update_data["assigned_to"] = assigned_to
   281	
   282	                await session.execute(
   283	                    TaskRecord.__table__.update()
   284	                    .where(TaskRecord.id == task_id)
   285	                    .values(**update_data)
   286	                )
   287	                await session.commit()
   288	
   289	                # Invalidate cache
   290	                cache_client = await self.cache_manager.get_client()
   291	                await cache_client.delete(f"project:{task_record.project_id}:actionable_tasks")
   292	
   293	                # Get next suggested actions for ADHD accommodation
   294	                next_actions = await self.get_next_actionable_tasks(task_record.project_id, 3)
   295	
   296	                response = {
   297	                    "success": True,
   298	                    "task_id": task_id,
   299	                    "old_status": old_status,
   300	                    "new_status": new_status.value,
   301	                    "instance": settings.instance_name,
   302	                    "timestamp": datetime.utcnow().isoformat(),
   303	                    "suggested_next_actions": [
   304	                        {"id": t.id, "title": t.title, "priority": t.priority.value}
   305	                        for t in next_actions
   306	                    ]
   307	                }
   308	
   309	                if new_status == TaskStatus.COMPLETED:
   310	                    logger.info(f"✅ Task {task_id} completed!")
   311	
   312	                return response
   313	
   314	        except Exception as e:
   315	            logger.error(f"❌ Task status update failed: {e}")
   316	            raise
