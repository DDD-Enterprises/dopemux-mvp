    48	    def _resolve_semantic_search_tool(self) -> Any:
    49	        """
    50	        Resolve semantic-search MCP tool with compatibility fallback.
    51	
    52	        Preferred:
    53	        - mcp__conport__semantic_search
    54	        Legacy fallback:
    55	        - mcp__conport__semantic_search_conport
    56	        """
    57	        preferred = getattr(self.mcp_tools, "mcp__conport__semantic_search", None)
    58	        if callable(preferred):
    59	            return preferred, "mcp__conport__semantic_search"
    60	
    61	        legacy = getattr(self.mcp_tools, "mcp__conport__semantic_search_conport", None)
    62	        if callable(legacy):
    63	            return legacy, "mcp__conport__semantic_search_conport"
    64	
    65	        raise AttributeError(
    66	            "No ConPort semantic-search MCP tool found "
    67	            "(expected mcp__conport__semantic_search or mcp__conport__semantic_search_conport)"
    68	        )

    70	    async def log_progress(
    71	        self,
    72	        workspace_id: str,
    73	        status: str,
    74	        description: str,
    75	        tags: Optional[List[str]] = None,
    76	        linked_item_type: Optional[str] = None,
    77	        linked_item_id: Optional[str] = None,
    78	        link_relationship_type: Optional[str] = None
    79	    ) -> Dict[str, Any]:
    80	        """
    81	        Log progress entry to ConPort.
    82	
    83	        Args:
    84	            workspace_id: Absolute path to workspace
    85	            status: TODO, IN_PROGRESS, DONE, BLOCKED
    86	            description: Task description
    87	            tags: Optional list of tags
    88	            linked_item_type: Optional type of linked item
    89	            linked_item_id: Optional ID of linked item
    90	            link_relationship_type: Optional relationship type
    91	
    92	        Returns:
    93	            Dict with {"id": int, ...} on success
    94	
    95	        Raises:
    96	            Exception: If MCP call fails
    97	        """
    98	        try:
    99	            params = {
   100	                "workspace_id": workspace_id,
   101	                "status": status,
   102	                "description": description
   103	            }
   104	
   105	            if tags:
   106	                params["tags"] = tags
   107	
   108	            if linked_item_type and linked_item_id:
   109	                params["linked_item_type"] = linked_item_type
   110	                params["linked_item_id"] = linked_item_id
   111	
   112	            if link_relationship_type:
   113	                params["link_relationship_type"] = link_relationship_type
   114	
   115	            # Call ConPort MCP tool
   116	            result = await self.mcp_tools.mcp__conport__log_progress(**params)
   117	
   118	            logger.debug(f"ConPort log_progress: {result}")
   119	            return result
   120	
   121	        except Exception as e:
   122	            logger.error(f"ConPort log_progress failed: {e}")
   123	            raise

   491	    async def semantic_search(
   492	        self,
   493	        workspace_id: str,
   494	        query_text: str,
   495	        top_k: int = 5,
   496	        filter_item_types: Optional[List[str]] = None,
   497	        filter_tags_include_any: Optional[List[str]] = None,
   498	        filter_tags_include_all: Optional[List[str]] = None
   499	    ) -> Dict[str, Any]:
   500	        """
   501	        Semantic search across ConPort items.
   502	
   503	        Args:
   504	            workspace_id: Absolute path to workspace
   505	            query_text: Natural language query
   506	            top_k: Number of results (default 5, max 25)
   507	            filter_item_types: Optional filter by item types
   508	            filter_tags_include_any: Optional filter by tags (any)
   509	            filter_tags_include_all: Optional filter by tags (all)
   510	
   511	        Returns:
   512	            Dict with search results
   513	
   514	        Raises:
   515	            Exception: If MCP call fails
   516	        """
   517	        try:
   518	            params = {
   519	                "workspace_id": workspace_id,
   520	                "query_text": query_text,
   521	                "top_k": str(min(top_k, 25))  # Enforce max 25 as string for MCP validation
   522	            }
   523	
   524	            if filter_item_types:
   525	                params["filter_item_types"] = filter_item_types
   526	
   527	            if filter_tags_include_any:
   528	                params["filter_tags_include_any"] = filter_tags_include_any
   529	
   530	            if filter_tags_include_all:
   531	                params["filter_tags_include_all"] = filter_tags_include_all
   532	
   533	            tool, tool_name = self._resolve_semantic_search_tool()
   534	            result = await tool(**params)
   535	
   536	            if isinstance(result, dict):
   537	                result.setdefault("_tool_used", tool_name)
   538	                result_count = len(result.get("results", []))
   539	            else:
   540	                result_count = 0
   541	            logger.debug(
   542	                "ConPort semantic_search via %s: '%s' -> %s results",
   543	                tool_name,
   544	                query_text,
   545	                result_count,
   546	            )
   547	            return result
   548	
   549	        except Exception as e:
   550	            logger.error(f"ConPort semantic_search failed: {e}")
   551	            raise

   748	    async def batch_log_progress(
   749	        self,
   750	        workspace_id: str,
   751	        progress_entries: List[Dict[str, Any]],
   752	        return_exceptions: bool = True
   753	    ) -> List[Any]:
   754	        """
   755	        Log multiple progress entries in parallel.
   756	
   757	        Args:
   758	            workspace_id: Workspace identifier
   759	            progress_entries: List of progress entry dicts (each with status, description, etc.)
   760	            return_exceptions: If True, exceptions are returned instead of raised
   761	
   762	        Returns:
   763	            List of results, one per progress entry
   764	
   765	        Example:
   766	            entries = [
   767	                {"status": "DONE", "description": "Task 1 complete"},
   768	                {"status": "IN_PROGRESS", "description": "Task 2 in progress"}
   769	            ]
   770	            results = await client.batch_log_progress(workspace_id, entries)
   771	        """
   772	        self.__init_parallel_components()
   773	
   774	        # Add workspace_id to each entry
   775	        call_specs = []
   776	        for entry in progress_entries:
   777	            entry_with_workspace = {**entry, "workspace_id": workspace_id}
   778	            call_specs.append({
   779	                "method": "log_progress",
   780	                "kwargs": entry_with_workspace
   781	            })
   782	
   783	        logger.info(f"Batch logging {len(progress_entries)} progress entries in parallel")
   784	        return await self._parallel_executor.execute_batch(self, call_specs, return_exceptions)
