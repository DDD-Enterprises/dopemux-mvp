    56	    async def handle_request(self, request: Dict[str, Any]) -> Dict[str, Any]:
    57	        """Handle MCP requests."""
    58	        if request.get("type") == "call_tool":
    59	            tool_name = request.get("name")
    60	            args = request.get("arguments", {})
    61	
    62	            if tool_name == "search_code":
    63	                return await self.search_code(args)
    64	            elif tool_name == "docs_search":
    65	                return await self.docs_search(args)
    66	            elif tool_name == "get_index_status":
    67	                return await self.get_index_status()
    68	            else:
    69	                return {"error": f"Unknown tool: {tool_name}"}
    70	
    71	        elif request.get("type") == "list_tools":
    72	            return {"tools": self.tools}
    73	
    74	        else:
    75	            return {"error": "Unknown request type"}
