     1	#!/usr/bin/env python3
     2	"""
     3	Task-Master-AI MCP Server Wrapper
     4	
     5	This wrapper provides an MCP server interface for Task-Master-AI,
     6	which handles PRD parsing and task decomposition with ADHD accommodations.
     7	
     8	The wrapper can either:
     9	1. Launch the external task-master-ai package via uvx
    10	2. Act as a stdio proxy with event emission capabilities
    11	"""
    12	
    13	import asyncio
    14	import json
    15	import sys
    16	import os
    17	import subprocess
    18	import logging
    19	from pathlib import Path
    20	from typing import Optional, Dict, Any
    21	from datetime import datetime
    22	
    23	# Add dopemux to path for event bus integration
    24	project_root = Path(__file__).parent.parent.parent
    25	sys.path.insert(0, str(project_root))
    26	
    27	try:
    28	    from dopemux.event_bus import RedisStreamsAdapter, DopemuxEvent, Priority, CognitiveLoad, ADHDMetadata
    29	    from dopemux.producers.mcp_producer import MCPEventProducer
    30	    EVENTS_ENABLED = True
    31	except ImportError:
    32	    EVENTS_ENABLED = False
    33	    logging.warning("Event bus integration not available - continuing without events")
    34	
    35	# Configure logging
    36	logging.basicConfig(
    37	    level=logging.INFO,
    38	    format='%(asctime)s - %(name)s - %(levelname)s - %(message)s',
    39	    stream=sys.stderr
    40	)
    41	logger = logging.getLogger("task-master-wrapper")
    42	
    43	
    44	class TaskMasterWrapper:
    45	    """
    46	    MCP Wrapper for Task-Master-AI with ADHD accommodations.
    47	
    48	    This wrapper provides:
    49	    - Transparent stdio proxy to the underlying task-master-ai server
    50	    - Event emission for task creation and updates
    51	    - ADHD-friendly progress tracking
    52	    - ConPort integration for decision tracking
    53	    """
    54	
    55	    def __init__(self):
    56	        self.process: Optional[subprocess.Popen] = None
    57	        self.event_bus: Optional[RedisStreamsAdapter] = None
    58	        self.mcp_producer: Optional[MCPEventProducer] = None
    59	        self.instance_id = os.getenv("DOPEMUX_INSTANCE", "default")
    60	        self.workspace_id = os.getenv("WORKSPACE_ID", "/Users/hue/code/dopemux-mvp")
    61	
    62	        # ADHD configuration
    63	        self.adhd_config = {
    64	            "task_chunk_size": int(os.getenv("TASK_CHUNK_SIZE", "5")),  # Max tasks per chunk
    65	            "focus_duration": int(os.getenv("FOCUS_DURATION", "25")),   # Minutes per session
    66	            "celebration_enabled": os.getenv("CELEBRATION_ENABLED", "true").lower() == "true",
    67	            "progress_visualization": os.getenv("PROGRESS_VIZ", "true").lower() == "true"
    68	        }
    69	
    70	        # Track active tasks for progress monitoring
    71	        self.active_tasks = {}
    72	        self.pending_calls = {}
    73	
    74	    async def initialize_event_bus(self):
    75	        """Initialize connection to Redis event bus if available."""
    76	        if not EVENTS_ENABLED:
    77	            return
    78	
    79	        try:
    80	            redis_url = os.getenv("REDIS_URL", "redis://localhost:6379")
    81	            self.event_bus = RedisStreamsAdapter(redis_url)
    82	            await self.event_bus.connect()
    83	
    84	            self.mcp_producer = MCPEventProducer(
    85	                self.event_bus,
    86	                self.instance_id
    87	            )
    88	
    89	            logger.info(f"Connected to event bus for instance {self.instance_id}")
    90	
    91	        except Exception as e:
    92	            logger.warning(f"Failed to connect to event bus: {e}")
    93	            logger.info("Continuing without event emission")
    94	
    95	    async def start_task_master(self):
    96	        """Start the underlying Task-Master-AI server."""
    97	        try:
    98	            # Check if we should use uvx or npx
    99	            use_uvx = os.getenv("USE_UVX", "true").lower() == "true"
   100	
   101	            if use_uvx:
   102	                # Use uvx to run task-master-ai
   103	                cmd = ["uvx", "task-master-ai"]
   104	            else:
   105	                # Use npx for the Node.js version
   106	                cmd = ["npx", "claude-task-master"]
   107	
   108	            # Add any additional environment variables
   109	            env = os.environ.copy()
   110	            env["OPENAI_API_KEY"] = os.getenv("OPENAI_API_KEY", "")
   111	            env["ANTHROPIC_API_KEY"] = os.getenv("ANTHROPIC_API_KEY", "")
   112	
   113	            # Start the process
   114	            self.process = subprocess.Popen(
   115	                cmd,
   116	                stdin=subprocess.PIPE,
   117	                stdout=subprocess.PIPE,
   118	                stderr=subprocess.PIPE,
   119	                env=env,
   120	                text=False  # Use binary mode for proper stdio handling
   121	            )
   122	
   123	            logger.info(f"Started Task-Master-AI server with command: {' '.join(cmd)}")
   124	
   125	        except FileNotFoundError as e:
   126	            logger.error(f"Failed to start Task-Master-AI: {e}")
   127	            logger.error("Please install task-master-ai via: uvx install task-master-ai")
   128	            sys.exit(1)
   129	        except Exception as e:
   130	            logger.error(f"Unexpected error starting Task-Master-AI: {e}")
   131	            sys.exit(1)
   132	
   133	    async def emit_task_event(self, event_type: str, data: Dict[str, Any]):
   134	        """Emit a task-related event to the event bus."""
   135	        if not self.mcp_producer:
   136	            return
   137	
   138	        try:
   139	            event = DopemuxEvent(
   140	                event_type=f"task.{event_type}",
   141	                source=f"task-master-{self.instance_id}",
   142	                instance_id=self.instance_id,
   143	                data=data,
   144	                priority=Priority.MEDIUM,
   145	                cognitive_load=CognitiveLoad.MEDIUM,
   146	                adhd_metadata=ADHDMetadata(
   147	                    interruption_allowed=event_type != "in_progress",
   148	                    focus_required=event_type == "in_progress",
   149	                    time_sensitive=False,
   150	                    can_batch=True
   151	                )
   152	            )
   153	
   154	            await self.event_bus.publish(event)
   155	            logger.debug(f"Emitted {event_type} event for task")
   156	
   157	        except Exception as e:
   158	            logger.warning(f"Failed to emit event: {e}")
   159	
   160	    async def handle_message(self, message: bytes) -> Optional[bytes]:
   161	        """
   162	        Process a message from Claude Code, optionally emit events,
   163	        and forward to the Task-Master-AI server.
   164	        """
   165	        try:
   166	            # Try to parse as JSON-RPC
   167	            try:
   168	                msg_str = message.decode('utf-8')
   169	                msg_json = json.loads(msg_str)
   170	
   171	                # Track method calls for event emission
   172	                if msg_json.get("method") == "tools/call":
   173	                    params = msg_json.get("params", {})
   174	                    tool_name = params.get("name", "")
   175	
   176	                    # Store pending call for response tracking
   177	                    msg_id = msg_json.get("id")
   178	                    if msg_id:
   179	                        self.pending_calls[msg_id] = {
   180	                            "tool": tool_name,
   181	                            "params": params,
   182	                            "start_time": datetime.now()
   183	                        }
   184	
   185	                    # Emit event for task creation
   186	                    if tool_name in ["create_task", "parse_prd", "decompose_task"]:
   187	                        await self.emit_task_event("created", {
   188	                            "tool": tool_name,
   189	                            "params": params,
   190	                            "timestamp": datetime.now().isoformat()
   191	                        })
   192	
   193	                        # Show ADHD-friendly progress if enabled
   194	                        if self.adhd_config["progress_visualization"]:
   195	                            progress_msg = self._generate_progress_message(tool_name)
   196	                            if progress_msg:
   197	                                sys.stderr.write(f"\n{progress_msg}\n")
   198	                                sys.stderr.flush()
   199	
   200	            except (json.JSONDecodeError, UnicodeDecodeError):
   201	                # Not JSON, just pass through
   202	                pass
   203	
   204	            # Forward to Task-Master-AI
   205	            if self.process and self.process.stdin:
   206	                self.process.stdin.write(message)
   207	                self.process.stdin.flush()
   208	
   209	        except Exception as e:
   210	            logger.error(f"Error handling message: {e}")
   211	
   212	    async def handle_response(self, response: bytes) -> Optional[bytes]:
   213	        """
   214	        Process a response from Task-Master-AI, optionally emit events,
   215	        and forward to Claude Code.
   216	        """
   217	        try:
   218	            # Try to parse as JSON-RPC response
   219	            try:
   220	                resp_str = response.decode('utf-8')
   221	                resp_json = json.loads(resp_str)
   222	
   223	                # Check if this is a response to a tracked call
   224	                msg_id = resp_json.get("id")
   225	                if msg_id and msg_id in self.pending_calls:
   226	                    call_info = self.pending_calls.pop(msg_id)
   227	                    duration = (datetime.now() - call_info["start_time"]).total_seconds()
   228	
   229	                    # Emit completion event
   230	                    await self.emit_task_event("completed", {
   231	                        "tool": call_info["tool"],
   232	                        "duration": duration,
   233	                        "success": "error" not in resp_json,
   234	                        "timestamp": datetime.now().isoformat()
   235	                    })
   236	
   237	                    # Celebrate completion if enabled
   238	                    if self.adhd_config["celebration_enabled"] and "error" not in resp_json:
   239	                        celebration = self._generate_celebration(call_info["tool"])
   240	                        if celebration:
   241	                            sys.stderr.write(f"\n{celebration}\n")
   242	                            sys.stderr.flush()
   243	
   244	            except (json.JSONDecodeError, UnicodeDecodeError):
   245	                # Not JSON, just pass through
   246	                pass
   247	
   248	            # Forward to Claude Code
   249	            sys.stdout.buffer.write(response)
   250	            sys.stdout.buffer.flush()
   251	
   252	        except Exception as e:
   253	            logger.error(f"Error handling response: {e}")
   254	
   255	    def _generate_progress_message(self, tool_name: str) -> str:
   256	        """Generate ADHD-friendly progress messages."""
   257	        messages = {
   258	            "create_task": "🎯 Creating new task...",
   259	            "parse_prd": "📋 Parsing PRD document...",
   260	            "decompose_task": "🔍 Breaking down into subtasks...",
   261	            "update_task": "✏️ Updating task details...",
   262	            "complete_task": "✅ Marking task complete..."
   263	        }
   264	        return messages.get(tool_name, f"⚡ Processing {tool_name}...")
   265	
   266	    def _generate_celebration(self, tool_name: str) -> str:
   267	        """Generate ADHD-friendly celebration messages."""
   268	        celebrations = {
   269	            "create_task": "✨ Task created successfully!",
   270	            "parse_prd": "🎉 PRD parsed and understood!",
   271	            "decompose_task": "🌟 Task breakdown complete!",
   272	            "update_task": "💫 Task updated!",
   273	            "complete_task": "🎊 Task completed - great work!"
   274	        }
   275	        return celebrations.get(tool_name, "✅ Operation successful!")
   276	
   277	    async def run(self):
   278	        """Main run loop for the wrapper."""
   279	        # Initialize event bus
   280	        await self.initialize_event_bus()
   281	
   282	        # Start Task-Master-AI server
   283	        await self.start_task_master()
   284	
   285	        # Create tasks for reading from stdin and Task-Master-AI
   286	        tasks = []
   287	
   288	        # Task to read from stdin and forward to Task-Master-AI
   289	        async def read_stdin():
   290	            loop = asyncio.get_event_loop()
   291	            while True:
   292	                try:
   293	                    # Read from stdin in non-blocking mode
   294	                    message = await loop.run_in_executor(None, sys.stdin.buffer.read1, 8192)
   295	                    if message:
   296	                        await self.handle_message(message)
   297	                    else:
   298	                        # EOF reached
   299	                        break
   300	                except Exception as e:
   301	                    logger.error(f"Error reading stdin: {e}")
   302	                    break
   303	
   304	        # Task to read from Task-Master-AI and forward to stdout
   305	        async def read_task_master():
   306	            loop = asyncio.get_event_loop()
   307	            while self.process and self.process.stdout:
   308	                try:
   309	                    # Read from Task-Master-AI in non-blocking mode
   310	                    response = await loop.run_in_executor(None, self.process.stdout.read1, 8192)
   311	                    if response:
   312	                        await self.handle_response(response)
   313	                    else:
   314	                        # EOF reached
   315	                        break
   316	                except Exception as e:
   317	                    logger.error(f"Error reading from Task-Master-AI: {e}")
   318	                    break
   319	
   320	        # Start both tasks
   321	        tasks.append(asyncio.create_task(read_stdin()))
   322	        tasks.append(asyncio.create_task(read_task_master()))
   323	
   324	        # Also forward stderr
   325	        async def read_stderr():
   326	            loop = asyncio.get_event_loop()
   327	            while self.process and self.process.stderr:
   328	                try:
   329	                    error = await loop.run_in_executor(None, self.process.stderr.read1, 8192)
   330	                    if error:
   331	                        sys.stderr.buffer.write(error)
   332	                        sys.stderr.buffer.flush()
   333	                    else:
   334	                        break
   335	                except Exception as e:
   336	                    logger.error(f"Error reading stderr: {e}")
   337	                    break
   338	
   339	        tasks.append(asyncio.create_task(read_stderr()))
   340	
   341	        # Wait for any task to complete (usually stdin EOF)
   342	        done, pending = await asyncio.wait(tasks, return_when=asyncio.FIRST_COMPLETED)
   343	
   344	        # Cancel remaining tasks
   345	        for task in pending:
   346	            task.cancel()
   347	
   348	        # Clean up
   349	        if self.process:
   350	            self.process.terminate()
   351	            await asyncio.sleep(0.5)
   352	            if self.process.poll() is None:
   353	                self.process.kill()
   354	
   355	        if self.event_bus:
   356	            await self.event_bus.disconnect()
   357	
   358	        logger.info("Task-Master-AI wrapper shut down")
   359	
   360	
   361	async def main():
   362	    """Main entry point."""
   363	    wrapper = TaskMasterWrapper()
   364	    try:
   365	        await wrapper.run()
   366	    except KeyboardInterrupt:
   367	        logger.info("Received interrupt, shutting down...")
   368	    except Exception as e:
   369	        logger.error(f"Fatal error: {e}")
   370	        sys.exit(1)
   371	
   372	
   373	if __name__ == "__main__":
   374	    asyncio.run(main())
