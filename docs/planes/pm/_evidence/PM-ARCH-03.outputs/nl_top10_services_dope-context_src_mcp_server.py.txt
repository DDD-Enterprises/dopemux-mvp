   155	async def get_dynamic_top_k(user_id: str = "default", requested_top_k: int = 10) -> int:
   156	    """
   157	    Get dynamic top_k from ADHD Engine based on user's attention state.
   158	
   159	    Returns:
   160	        scattered: 5 results
   161	        focused: 15 results
   162	        hyperfocused: 40 results
   163	        fallback: requested_top_k (original behavior)
   164	    """
   165	    adhd_config, feature_flags = await _get_adhd_config()
   166	
   167	    if adhd_config and feature_flags:
   168	        try:
   169	            from feature_flags import FEATURE_ADHD_ENGINE_DOPE_CONTEXT
   170	
   171	            if await feature_flags.is_enabled(
   172	                FEATURE_ADHD_ENGINE_DOPE_CONTEXT,
   173	                "dope-context",
   174	                user_id
   175	            ):
   176	                return await adhd_config.get_max_results(user_id)
   177	        except Exception as e:
   178	            logger.error(f"ADHD Engine query failed: {e}")
   179	
   180	    # Fallback to requested value
   181	    return requested_top_k

   318	def _load_snapshot_metadata(workspace: Path, workspace_hash: str) -> Dict[str, Any]:
   319	    """
   320	    Load stored snapshot metadata (if available) for workspace.
   321	
   322	    Returns:
   323	        Dictionary with optional keys: files_indexed, total_chunks, last_snapshot
   324	    """
   325	    snapshot_dir = _snapshot_dir_for_hash(workspace_hash)
   326	    metadata: Dict[str, Any] = {
   327	        "snapshot_dir": str(snapshot_dir),
   328	    }
   329	
   330	    snapshot_file = snapshot_dir / "snapshot.json"
   331	    if snapshot_file.exists():
   332	        try:
   333	            data = json.loads(snapshot_file.read_text())
   334	            metadata["files_indexed"] = len(data.get("files", {}))
   335	            metadata["last_snapshot"] = data.get("created_at") or datetime.fromtimestamp(
   336	                snapshot_file.stat().st_mtime
   337	            ).isoformat()
   338	        except Exception as exc:
   339	            logger.debug("Failed to parse snapshot %s: %s", snapshot_file, exc)
   340	
   341	    chunk_snapshot_file = snapshot_dir / "chunk_snapshot.json"
   342	    if chunk_snapshot_file.exists():
   343	        try:
   344	            data = json.loads(chunk_snapshot_file.read_text())
   345	            files = data.get("files", {})
   346	            metadata["total_chunks"] = sum(len(file.get("chunks", [])) for file in files.values())
   347	        except Exception as exc:
   348	            logger.debug("Failed to parse chunk snapshot %s: %s", chunk_snapshot_file, exc)
   349	
   350	    metadata["workspace_exists"] = workspace.exists()
   351	    return metadata

   354	def _discover_workspaces_from_snapshots() -> List[Path]:
   355	    """
   356	    Discover previously indexed workspaces by reading snapshot metadata.
   357	
   358	    Returns:
   359	        List of workspace paths (may include paths that no longer exist)
   360	    """
   361	    base = _snapshot_root()
   362	    if not base.exists():
   363	        return []
   364	
   365	    discovered: List[Path] = []
   366	    for child in sorted(base.iterdir()):
   367	        if not child.is_dir():
   368	            continue
   369	
   370	        workspace_path: Optional[Path] = None
   371	        for candidate in ("snapshot.json", "chunk_snapshot.json"):
   372	            file_path = child / candidate
   373	            if not file_path.exists():
   374	                continue
   375	            try:
   376	                data = json.loads(file_path.read_text())
   377	                workspace_str = data.get("workspace_path")
   378	            except Exception:
   379	                continue
   380	            if workspace_str:
   381	                workspace_path = Path(workspace_str)
   382	                break
   383	
   384	        if workspace_path:
   385	            discovered.append(workspace_path)
   386	
   387	    return discovered

   390	def _resolve_target_workspaces(
   391	    workspace_path: Optional[str] = None,
   392	    workspace_paths: Optional[List[str]] = None,
   393	) -> List[Path]:
   394	    """
   395	    Resolve a unique ordered list of workspaces to operate on.
   396	    """
   397	    candidates: List[Path] = []
   398	
   399	    if workspace_paths:
   400	        candidates.extend(Path(p).expanduser() for p in workspace_paths if p)
   401	    if workspace_path:
   402	        candidates.append(Path(workspace_path).expanduser())
   403	
   404	    if not candidates:
   405	        discovered = _discover_workspaces_from_snapshots()
   406	        if discovered:
   407	            candidates.extend(discovered)
   408	        else:
   409	            candidates.append(get_workspace_root())
   410	
   411	    resolved: List[Path] = []
   412	    seen: Set[str] = set()
   413	    for candidate in candidates:
   414	        resolved_path = candidate.resolve()
   415	        key = str(resolved_path)
   416	        if key not in seen:
   417	            resolved.append(resolved_path)
   418	            seen.add(key)
   419	
   420	    return resolved

   423	def _resolve_explicit_workspaces(
   424	    workspace_path: Optional[str] = None,
   425	    workspace_paths: Optional[List[str]] = None,
   426	    *,
   427	    fallback_to_current: bool = True,
   428	) -> List[Path]:
   429	    """
   430	    Resolve explicitly requested workspaces (without snapshot discovery).
   431	    """
   432	    candidates: List[Path] = []
   433	
   434	    if workspace_paths:
   435	        candidates.extend(Path(p).expanduser() for p in workspace_paths if p)
   436	    if workspace_path:
   437	        candidates.append(Path(workspace_path).expanduser())
   438	
   439	    if not candidates and fallback_to_current:
   440	        candidates.append(get_workspace_root())
   441	
   442	    resolved: List[Path] = []
   443	    seen: Set[str] = set()
   444	    for candidate in candidates:
   445	        resolved_path = candidate.resolve()
   446	        key = str(resolved_path)
   447	        if key not in seen:
   448	            resolved.append(resolved_path)
   449	            seen.add(key)
   450	
   451	    return resolved

   455	    """
   456	    Fetch collection information from Qdrant with graceful error handling.
   457	    """

   556	        chunker=chunker,

   558	        standard_embedder=_embedder,

   563	
   564	    # Initialize docs components (contextualized embeddings)

   566	        api_key=voyage_key,
   567	        cache_ttl_hours=24,

   576	    _docs_pipeline = DocIndexingPipeline(
   577	        embedder=_docs_embedder,
   578	        doc_search=_docs_search,
   579	        workspace_path=Path.cwd(),
   580	        workspace_id=os.getenv("WORKSPACE_ID", "default"),
   581	    )
   582	
   583	    logger.info("Dope-Context MCP server initialized (code + docs)")

   627	        workspace_path=workspace,
   628	        include_patterns=include_patterns or ["*.py", "*.js", "*.ts", "*.tsx"],
   629	        exclude_patterns=exclude_patterns or ["*test*", "*__pycache__*"],
   630	        max_files=max_files,
   631	        workspace_id=str(workspace),
   632	    )
   633	
   634	    pipeline = IndexingPipeline(
   635	        chunker=chunker,
   636	        context_generator=context_generator,
   637	        standard_embedder=standard_embedder,
   638	        contextualized_embedder=contextualized_embedder,
   639	        vector_search=vector_search,
   640	        config=config,
   641	    )
   642	
   643	    # Run indexing
   644	    progress = await pipeline.index_workspace()
   645	
   646	    # Build BM25 index for hybrid search (after vector indexing completes)
   647	    try:
   648	        logger.info("Building BM25 index for keyword search...")

   661	                    'bm25': bm25_index.bm25,
   662	                    'documents': bm25_index.documents,
   663	                    'doc_ids': bm25_index.doc_ids,

  1804

  1817	            "has_changes": False,
  1818	            "total_changes": 0,
  1819	            "changes": 0,
  1820	            "added": 0,
  1821	            "modified": 0,
  1822	            "removed": 0,
  1823	            "message": "No changes detected",
  1824	        }
  1825	
  1826	    # Incremental reindexing if requested
  1827	    if auto_reindex:
  1828	        logger.info(f"Auto-reindexing {changes.total_changes()} changed files...")
  1829	
  1830	        try:
  1831	            # Get cached components
  1832	            qdrant_url = os.getenv("QDRANT_URL", "localhost")
  1833	            qdrant_port = int(os.getenv("QDRANT_PORT", "6333"))

  1838	            openai_key = os.getenv("OPENAI_API_KEY")
  1839	            context_generator = OpenAIContextGenerator(api_key=openai_key) if openai_key else None

  1841	            standard_embedder = _get_cached_embedder(api_key=voyage_key)
  1842	            contextualized_embedder = _get_cached_contextualized_embedder(api_key=voyage_key)
  1843	
  1844	            config = IndexingConfig(
  1845	                workspace_path=workspace,
  1846	                include_patterns=include_patterns or ["*.py", "*.js", "*.ts", "*.tsx"],
  1847	                workspace_id=str(workspace),
  1848	            )
  1849	
  1850	            pipeline = IndexingPipeline(
  1851	                chunker=chunker,
  1852	                context_generator=context_generator,
  1853	                standard_embedder=standard_embedder,
  1854	                contextualized_embedder=contextualized_embedder,
  1855	                vector_search=vector_search,
  1856	                config=config,
  1857	            )
  1858	
  1859	            # Index only changed files
  1860	            changed_files = changes.added + changes.modified
  1861	            if changed_files:
  1862	                changed_patterns = [Path(path).as_posix().lstrip("./") for path in changed_files]
  1863	                # Create temporary config with only changed files
  1864	                temp_config = IndexingConfig(
  1865	                    workspace_path=workspace,
  1866	                    include_patterns=changed_patterns,  # Only these specific files
  1867	                    workspace_id=str(workspace),
  1868	                )
  1869	                pipeline.config = temp_config
  1870	                await pipeline.index_workspace()
  1871	                logger.info(f"Reindexed {len(changed_files)} added/modified files")
  1872	
  1873	            # Delete removed files from collection
  1874	            if changes.removed:
  1875	                all_docs = await vector_search.get_all_payloads()
  1876	                removed_set = {Path(path).as_posix() for path in changes.removed}
  1877	                removed_basenames = {Path(path).name for path in changes.removed}
  1878	                point_ids_to_delete: List[str] = []
  1879	                for doc in all_docs:
  1880	                    source_path = doc.get("file_path") or doc.get("source_path")
  1881	                    point_id = doc.get("id")
  1882	                    if not source_path or not point_id:
  1883	                        continue
  1884	                    source_norm = Path(str(source_path)).as_posix()
  1885	                    if (
  1886	                        source_norm in removed_set
  1887	                        or source_norm.lstrip("./") in removed_set
  1888	                        or Path(source_norm).name in removed_basenames
  1889	                    ):
  1890	                        point_ids_to_delete.append(str(point_id))
  1891	                if point_ids_to_delete:
  1892	                    await vector_search.delete_points(point_ids_to_delete)
  1893	                    logger.info(
  1894	                        "Deleted %s vectors for %s removed files",
  1895	                        len(point_ids_to_delete),
  1896	                        len(changes.removed),
  1897	                    )
  1898	                else:
  1899	                    logger.info(
  1900	                        "No matching vector points found for %s removed files",
  1901	                        len(changes.removed),
  1902	                    )
  1903	
  1904	            # Rebuild BM25 index after incremental changes
  1905	            try:
  1906	                all_docs = await vector_search.get_all_payloads()
  1907	                if all_docs:
  1908	                    bm25_index = BM25Index()
  1909	                    bm25_index.build_index(all_docs)
  1910	                    bm25_cache_path = get_snapshot_dir(workspace) / "bm25_index.pkl"
  1911	                    with open(bm25_cache_path, 'wb') as f:
  1912	                        pickle.dump({
  1913	                            'bm25': bm25_index.bm25,
  1914	                            'documents': bm25_index.documents,
  1915	                            'doc_ids': bm25_index.doc_ids,
  1916	                        }, f)
  1917	                    logger.info("BM25 index updated after sync")
  1918	            except Exception as bm25_error:
  1919	                logger.warning(f"BM25 update failed: {bm25_error}")
  1920	
  1921	            return {
  1922	                "workspace": str(workspace),
  1923	                "has_changes": changes.total_changes() > 0,
  1924	                "total_changes": changes.total_changes(),
  1925	                "changes": changes.total_changes(),
  1926	                "added": len(changes.added),
  1927	                "modified": len(changes.modified),
  1928	                "removed": len(changes.removed),
  1929	                "reindexed": len(changed_files),
  1930	                "message": f"Synced and reindexed {len(changed_files)} files",
  1931	            }
  1932	
  1933	        except Exception as reindex_error:
  1934	            logger.error(f"Auto-reindex failed: {reindex_error}")
  1935	            return {
  1936	                "workspace": str(workspace),
  1937	                "has_changes": changes.total_changes() > 0,
  1938	                "total_changes": changes.total_changes(),
  1939	                "changes": changes.total_changes(),
  1940	                "added": len(changes.added),
  1941	                "modified": len(changes.modified),
  1942	                "removed": len(changes.removed),
  1943	                "error": f"Sync detected changes but reindexing failed: {str(reindex_error)}",
  1944	                "message": f"Detected {changes.total_changes()} changes. Run index_workspace manually.",
  1945	            }
  1946	
  1947	    # Just report changes without reindexing
  1948	    return {
  1949	        "workspace": str(workspace),
  1950	        "has_changes": changes.total_changes() > 0,
  1951	        "total_changes": changes.total_changes(),
  1952	        "changes": changes.total_changes(),
  1953	        "added": len(changes.added),
  1954	        "modified": len(changes.modified),
  1955	        "removed": len(changes.removed),

  2533	@mcp.tool()
  2534	async def get_chunk_complexity(file_path: str, symbol: str) -> Dict:
  2535	    """
  2536	    Get AST-based complexity for code chunk (Synergy A support).
  2537	
  2538	    Returns complexity score 0.0-1.0 from Tree-sitter analysis.
  2539	    """
  2540	    resolved = Path(file_path).expanduser()
  2541	    if not resolved.is_absolute():
  2542	        resolved = (Path.cwd() / resolved).resolve()
  2543	    if not resolved.exists():
  2544	        return {
  2545	            "complexity": 0.5,
  2546	            "method": "ast-heuristic",
  2547	            "file_path": str(resolved),
  2548	            "symbol": symbol,
  2549	            "reason": "file_not_found",
  2550	        }
  2551	
  2552	    try:
  2553	        source = resolved.read_text(encoding="utf-8", errors="ignore")
  2554	    except Exception as exc:
  2555	        return {
  2556	            "complexity": 0.5,
  2557	            "method": "ast-heuristic",
  2558	            "file_path": str(resolved),
  2559	            "symbol": symbol,
  2560	            "reason": f"read_error:{exc}",
  2561	        }
  2562	
  2563	    try:
  2564	        tree = ast.parse(source)
  2565	    except SyntaxError:
  2566	        # Fallback heuristic for non-Python files or parse failures.
  2567	        line_count = max(1, len(source.splitlines()))
  2568	        keyword_hits = sum(
  2569	            source.lower().count(keyword)
  2570	            for keyword in ("if ", "for ", "while ", "try:", "except", "class ", "def ")
  2571	        )
  2572	        complexity = min(1.0, 0.15 + (line_count / 400.0) + (keyword_hits / 100.0))
  2573	        return {
  2574	            "complexity": round(complexity, 3),
  2575	            "method": "text-heuristic",
  2576	            "file_path": str(resolved),
  2577	            "symbol": symbol,
  2578	            "line_count": line_count,
  2579	            "keyword_hits": keyword_hits,
  2580	        }
  2581	
  2582	    symbol_node: Optional[ast.AST] = None
  2583	    for node in ast.walk(tree):
  2584	        if isinstance(node, (ast.FunctionDef, ast.AsyncFunctionDef, ast.ClassDef)):
  2585	            if getattr(node, "name", None) == symbol:
  2586	                symbol_node = node
  2587	                break
  2588	
  2589	    if symbol_node is not None:
  2590	        start = getattr(symbol_node, "lineno", 1)
  2591	        end = getattr(symbol_node, "end_lineno", start)
  2592	        target_source = "\n".join(source.splitlines()[start - 1 : end])
  2593	        target_tree = symbol_node
  2594	    else:
  2595	        start = 1
  2596	        end = len(source.splitlines())
  2597	        target_source = source
  2598	        target_tree = tree
  2599	
  2600	    branch_nodes = (
  2601	        ast.If,
  2602	        ast.For,
  2603	        ast.AsyncFor,
  2604	        ast.While,
  2605	        ast.Try,
  2606	        ast.With,
  2607	        ast.AsyncWith,
  2608	        ast.Match,
  2609	        ast.comprehension,
  2610	    )
  2611	    branch_count = sum(1 for node in ast.walk(target_tree) if isinstance(node, branch_nodes))
  2612	
  2613	    def _max_depth(node: ast.AST, depth: int = 0) -> int:
  2614	        children = list(ast.iter_child_nodes(node))
  2615	        if not children:
  2616	            return depth
  2617	        return max(_max_depth(child, depth + 1) for child in children)
  2618	
  2619	    max_depth = _max_depth(target_tree)
  2620	    line_count = max(1, len(target_source.splitlines()))
  2621	
  2622	    complexity = min(
  2623	        1.0,
  2624	        0.1 + (line_count / 220.0) + (branch_count / 20.0) + (max_depth / 30.0),
  2625	    )
  2626	    return {
  2627	        "complexity": round(complexity, 3),
  2628	        "method": "ast-heuristic",
  2629	        "file_path": str(resolved),
  2630	        "symbol": symbol,
  2631	        "line_count": line_count,
  2632	        "branch_count": branch_count,
  2633	        "max_depth": max_depth,
  2634	        "scope_start_line": start,
  2635	        "scope_end_line": end,
  2636	    }
  2637	
  2638	
  2639	if __name__ == "__main__":
