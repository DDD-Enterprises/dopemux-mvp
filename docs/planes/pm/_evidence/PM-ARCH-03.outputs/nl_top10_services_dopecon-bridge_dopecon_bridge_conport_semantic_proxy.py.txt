     1	"""
     2	Semantic-search proxy helpers for dopecon-bridge.
     3	
     4	Kept separate from main service wiring so fallback behavior can be unit-tested
     5	without importing the full bridge runtime stack.
     6	"""
     7	
     8	from __future__ import annotations
     9	
    10	import json
    11	from typing import Any, Dict, Optional
    12	
    13	
    14	async def run_semantic_search_with_fallback(
    15	    *,
    16	    session: Any,
    17	    base_url: str,
    18	    payload: Dict[str, Any],
    19	    logger: Any,
    20	) -> Dict[str, Any]:
    21	    """
    22	    Query ConPort semantic-search endpoints with compatibility fallback.
    23	
    24	    Primary endpoint: `/api/adhd/semantic-search`
    25	    Fallback endpoint: `/api/semantic-search` (only on 404/405 from primary)
    26	    """
    27	    endpoints = ["/api/adhd/semantic-search", "/api/semantic-search"]
    28	    last_error: Optional[str] = None
    29	
    30	    for index, endpoint in enumerate(endpoints):
    31	        url = f"{base_url}{endpoint}"
    32	        try:
    33	            async with session.post(url, json=payload) as response:
    34	                raw_text = await response.text()
    35	
    36	                if response.status == 200:
    37	                    if raw_text:
    38	                        try:
    39	                            body: Any = json.loads(raw_text)
    40	                        except json.JSONDecodeError:
    41	                            body = {"results": []}
    42	                    else:
    43	                        body = {"results": []}
    44	
    45	                    if isinstance(body, list):
    46	                        body = {"results": body}
    47	                    elif not isinstance(body, dict):
    48	                        body = {"results": []}
    49	
    50	                    if "results" not in body and isinstance(body.get("result"), list):
    51	                        body["results"] = body["result"]
    52	                    body.setdefault("results", [])
    53	                    body.setdefault("source", "conport")
    54	                    body.setdefault("endpoint", endpoint)
    55	                    return body
    56	
    57	                # Only fallback on endpoint-not-found style responses
    58	                if index == 0 and response.status in {404, 405}:
    59	                    logger.warning(
    60	                        "ConPort semantic search endpoint %s unavailable (%s); trying legacy endpoint",
    61	                        endpoint,
    62	                        response.status,
    63	                    )
    64	                    continue
    65	
    66	                last_error = f"{endpoint} returned {response.status}: {raw_text}"
    67	                break
    68	        except Exception as exc:
    69	            last_error = f"{endpoint} request error: {exc}"
    70	            if index == 0:
    71	                continue
    72	            break
    73	
    74	    raise RuntimeError(last_error or "ConPort semantic search failed")
