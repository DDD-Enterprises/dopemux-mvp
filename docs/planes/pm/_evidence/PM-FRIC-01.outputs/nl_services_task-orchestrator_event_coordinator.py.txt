   178	    async def _start_event_workers(self) -> None:
   179	        """Start event processing workers with priority handling."""
   180	        # Start workers for each priority level
   181	        priority_workers = {
   182	            EventPriority.CRITICAL: 3,   # 3 workers for critical events
   183	            EventPriority.HIGH: 2,       # 2 workers for high priority
   184	            EventPriority.MEDIUM: 2,     # 2 workers for medium priority
   185	            EventPriority.LOW: 1,        # 1 worker for low priority
   186	            EventPriority.BACKGROUND: 1  # 1 worker for background
   187	        }
   188	
   189	        for priority, worker_count in priority_workers.items():
   190	            for i in range(worker_count):
   191	                worker = asyncio.create_task(
   192	                    self._event_worker(priority, f"{priority.name.lower()}-{i}")
   193	                )
   194	                self.workers.append(worker)
   195	
   196	        logger.info(f"👥 Started {len(self.workers)} event processing workers")

   198	    async def _event_worker(self, priority: EventPriority, worker_id: str) -> None:
   199	        """Event processing worker for specific priority level."""
   200	        logger.debug(f"🚀 Event worker {worker_id} started")
   201	
   202	        while self.running:
   203	            try:
   204	                # Get event from priority queue
   205	                queue = self.event_queues[priority]
   206	                _, event = await asyncio.wait_for(queue.get(), timeout=5.0)
   207	
   208	                # Process event with ADHD considerations
   209	                await self._process_event(event, worker_id)
   210	
   211	            except asyncio.TimeoutError:
   212	                # No events at this priority, continue
   213	                continue
   214	            except Exception as e:
   215	                logger.error(f"Event worker {worker_id} error: {e}")
   216	                await asyncio.sleep(1.0)
   217	
   218	        logger.debug(f"🛑 Event worker {worker_id} stopped")

   220	    async def _process_event(self, event: CoordinationEvent, worker_id: str) -> None:
   221	        """Process individual coordination event."""
   222	        try:
   223	            start_time = datetime.now(timezone.utc)
   224	
   225	            # Apply ADHD filtering
   226	            if not await self._should_process_event(event):
   227	                self.metrics["events_filtered"] += 1
   228	                logger.debug(f"🎯 Event filtered (ADHD): {event.event_type}")
   229	                return
   230	
   231	            # Check if event has expired
   232	            if event.expires_at and datetime.now(timezone.utc) > event.expires_at:
   233	                logger.warning(f"⏰ Event expired: {event.id}")
   234	                return
   235	
   236	            # Get processors for this event type
   237	            processors = self.event_processors.get(event.event_type, [])
   238	
   239	            if not processors:
   240	                logger.warning(f"No processors registered for {event.event_type}")
   241	                return
   242	
   243	            # Execute processors in parallel
   244	            results = await asyncio.gather(
   245	                *[processor(event) for processor in processors],
   246	                return_exceptions=True
   247	            )
   248	
   249	            # Check for errors
   250	            errors = [r for r in results if isinstance(r, Exception)]
   251	            if errors:
   252	                logger.error(f"Event processing errors: {errors}")
   253	                event.error_message = str(errors[0])
   254	
   255	            # Update processing metadata
   256	            event.processed_at = datetime.now(timezone.utc)
   257	            processing_time = (event.processed_at - start_time).total_seconds()
   258	
   259	            # Store processing history in Redis
   260	            await self._store_event_history(event, processing_time)
   261	
   262	            self.metrics["events_processed"] += 1
   263	            logger.debug(f"✅ Event processed by {worker_id}: {event.event_type} in {processing_time:.3f}s")
   264	
   265	        except Exception as e:
   266	            logger.error(f"Event processing failed: {e}")
   267	            event.attempts += 1
   268	            event.error_message = str(e)
   269	
   270	            # Retry logic with exponential backoff
   271	            if event.attempts < event.max_attempts:
   272	                delay = 2 ** event.attempts  # Exponential backoff
   273	                logger.info(f"🔄 Retrying event {event.id} in {delay}s (attempt {event.attempts + 1})")
   274	
   275	                # Re-queue with delay
   276	                asyncio.create_task(self._delayed_requeue(event, delay))
   277	
   278	    async def _delayed_requeue(self, event: CoordinationEvent, delay: float) -> None:
   279	        """Re-queue event after delay."""
   280	        await asyncio.sleep(delay)
   281	        await self.emit_event(event)

   285	    async def emit_event(self, event: CoordinationEvent) -> bool:
   286	        """Emit coordination event for processing."""
   287	        try:
   288	            # Assign unique ID if not present
   289	            if not hasattr(event, 'id') or not event.id:
   290	                event.id = f"{event.event_type}_{datetime.now().timestamp()}"
   291	
   292	            # Apply ADHD filtering before queuing
   293	            if not await self._should_process_event(event):
   294	                return False
   295	
   296	            # Queue event by priority
   297	            priority_queue = self.event_queues[event.priority]
   298	            await priority_queue.put((event.priority.value, event))
   299	
   300	            logger.debug(f"📤 Event emitted: {event.event_type} (priority: {event.priority.name})")
   301	            return True
   302	
   303	        except Exception as e:
   304	            logger.error(f"Failed to emit event: {e}")
   305	            return False

   322	    async def _should_process_event(self, event: CoordinationEvent) -> bool:
   323	        """Determine if event should be processed based on ADHD state."""
   324	        try:
   325	            # Apply custom filters
   326	            for filter_func in self.event_filters:
   327	                if not await filter_func(event):
   328	                    return False
   329	
   330	            # Check focus mode restrictions
   331	            if self.current_focus_mode == "deep":
   332	                # In deep focus, only allow critical events
   333	                if event.priority not in [EventPriority.CRITICAL, EventPriority.HIGH]:
   334	                    return False
   335	
   336	                # Block interrupting events unless critical
   337	                if not event.interruption_allowed and event.priority != EventPriority.CRITICAL:
   338	                    return False
   339	
   340	            # Check energy level matching
   341	            if self.current_energy_level == "low":
   342	                # Low energy: avoid high cognitive load events
   343	                if event.cognitive_load > 0.7:
   344	                    return False
   345	
   346	            # Check for event flooding (ADHD protection)
   347	            recent_events = await self._get_recent_event_count(event.event_type)
   348	            if recent_events > 10:  # More than 10 similar events in last minute
   349	                logger.warning(f"🌊 Event flooding detected: {event.event_type}")
   350	                return False
   351	
   352	            return True
   353	
   354	        except Exception as e:
   355	            logger.error(f"Event filtering failed: {e}")
   356	            return True  # Default to processing on error

   831	    async def close(self) -> None:
   832	        """Shutdown event coordinator gracefully."""
   833	        logger.info("🛑 Shutting down Event Coordinator...")
   834	
   835	        # Stop processing
   836	        self.running = False
   837	
   838	        # Cancel workers
   839	        if self.workers:
   840	            for worker in self.workers:
   841	                worker.cancel()
   842	            await asyncio.gather(*self.workers, return_exceptions=True)
   843	
   844	        # Close Redis connection
   845	        if self.redis_client:
   846	            await self.redis_client.close()
   847	
   848	        logger.info("✅ Event Coordinator shutdown complete")
