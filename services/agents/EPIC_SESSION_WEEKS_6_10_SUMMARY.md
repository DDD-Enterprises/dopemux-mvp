### Complete End-to-End Example

```python
"""
Production workflow using ALL 7 agents
"""
from services.agents import (
    MemoryAgent,
    CognitiveGuardian,
    TwoPlaneOrchestrator,
    DopemuxEnforcer,
    ToolOrchestrator,
    TaskDecomposer,
    WorkflowCoordinator
)

async def production_feature_workflow(feature_description: str):
    workspace = "/Users/hue/code/dopemux-mvp"

    # 1. MemoryAgent - Context preservation
    memory = MemoryAgent(workspace_id=workspace)
    await memory.start_session(feature_description, complexity=0.7)
    # → Auto-saves every 30s, 0% context loss

    # 2. CognitiveGuardian - Break enforcement + energy matching
    guardian = CognitiveGuardian(workspace_id=workspace, memory_agent=memory)
    await guardian.start_monitoring()
    # → Reminds breaks at 25/60/90 min, prevents burnout

    # 3. TaskDecomposer - Break feature into tasks
    decomposer = TaskDecomposer()
    prd = load_prd(feature_description)
    tasks = await decomposer.decompose_prd(prd)
    # → 5-10 ADHD-friendly tasks with metadata

    # 4. WorkflowCoordinator - Orchestrate implementation
    workflow_coord = WorkflowCoordinator(workspace_id=workspace, memory_agent=memory)
    await workflow_coord.initialize()
    workflow = await workflow_coord.start_workflow(
        WorkflowType.FEATURE_IMPLEMENTATION,
        description=feature_description
    )
    # → 5-step workflow (design → implement → test → validate → document)

    # 5. ToolOrchestrator - Select optimal tools
    tool_selector = ToolOrchestrator(workspace_id=workspace)
    await tool_selector.initialize()

    template = workflow_coord.get_workflow_template(WorkflowType.FEATURE_IMPLEMENTATION)

    for step in template.steps:
        # Check readiness before each step
        readiness = await guardian.check_task_readiness(
            task_complexity=0.6,
            task_energy_required="medium"
        )

        if not readiness["ready"]:
            print(f"Not ready: {readiness['suggestion']}")
            break

        # Select optimal tools for step
        tools = await tool_selector.select_tools_for_task(
            task_type=step.step_type.value,
            complexity=0.6
        )
        # → Best MCP server + model for step

        # 6. TwoPlaneOrchestrator - Cross-plane coordination (if needed)
        orchestrator = TwoPlaneOrchestrator(workspace_id=workspace, bridge_url="http://localhost:3016")
        await orchestrator.initialize()

        # If step needs PM data:
        if step.step_type == StepType.DESIGN:
            tasks = await orchestrator.cognitive_to_pm("get_tasks", {})
            # → PM plane task data

        # Execute step with selected tools
        result = await execute_step_with_tools(step, tools)

        # 7. DopemuxEnforcer - Validate compliance
        enforcer = DopemuxEnforcer(workspace_id=workspace)
        await enforcer.initialize()

        if step.step_type == StepType.IMPLEMENT:
            compliance = await enforcer.validate_code_change(
                file_path=result.file_path,
                content=result.code
            )
            # → Gentle warnings if violations

            if not compliance.compliant:
                for violation in compliance.violations:
                    print(f"{violation.severity}: {violation.message}")

        # Checkpoint after step (WorkflowCoordinator + MemoryAgent)
        await workflow_coord.execute_step(
            workflow_id=workflow.workflow_id,
            step=step,
            result_data=result
        )
        # → Auto-saved, resumable

    # Complete workflow
    summary = await workflow_coord.complete_workflow(workflow.workflow_id)
    await memory.end_session()

    # ALL 7 AGENTS COORDINATED SEAMLESSLY!
    print(f"✅ Workflow complete: {summary['duration_minutes']} minutes")
```
