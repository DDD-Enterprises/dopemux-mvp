import os
import json
import re

def check_for_secrets(content):
    # Very basic secret detection regex
    secret_patterns = [
        r"(?i)api_key\s*[:=]\s*['\"][a-z0-9-_]{20,}['\"]",
        r"(?i)token\s*[:=]\s*['\"][a-z0-9-_]{20,}['\"]",
        r"(?i)secret\s*[:=]\s*['\"][a-z0-9-_]{20,}['\"]",
    ]
    for pattern in secret_patterns:
        if re.search(pattern, content):
            return True
    return False
