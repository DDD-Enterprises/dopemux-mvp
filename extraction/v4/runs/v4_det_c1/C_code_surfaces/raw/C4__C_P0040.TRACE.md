import asyncio
import logging
from typing import Dict, Any, List, Optional, Callable
from dataclasses import dataclass
from datetime import datetime

logger = logging.getLogger(__name__)

@dataclass
class DependencyStatus:
    status: str
    message: str
    details: Dict[str, Any]

async def check_dependency(
    name: str, 
    check_fn: Callable, 
    timeout_ms: int = 500, 
    critical: bool = False
) -> DependencyStatus:
    """Check a dependency health with timeout."""
    try:
        # Run check with timeout
        result = await asyncio.wait_for(check_fn(), timeout=timeout_ms / 1000.0)
        return DependencyStatus(
            status="ok" if result else "fail",
            message=f"{name} is {'up' if result else 'down'}",
            details={}
        )
    except asyncio.TimeoutError:
        return DependencyStatus(
            status="fail",
            message=f"{name} check timed out",
            details={"timeout_ms": timeout_ms}
        )
    except Exception as e:
        logger.error(f"Dependency check for {name} failed: {e}")
        return DependencyStatus(
            status="fail",
            message=str(e),
            details={"error": type(e).__name__}
        )

async def check_redis(redis_client) -> bool:
    """Check Redis connectivity."""
    if not redis_client:
        return False
    try:
        return await redis_client.ping()
    except Exception as e:
        logger.warning(f"Redis ping failed: {e}")
        return False
