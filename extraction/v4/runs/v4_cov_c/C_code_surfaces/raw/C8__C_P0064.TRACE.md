from dopemux.litellm_proxy import LiteLLMProxyInfo, LiteLLMProxyManager


def test_ensure_started_writes_config_and_master_key(monkeypatch, tmp_path):
    monkeypatch.setenv("OPENROUTER_API_KEY", "sk-provider")

    manager = LiteLLMProxyManager(tmp_path, "A", 3000)

    launches = {}

    monkeypatch.setattr(manager, "_is_port_in_use", lambda: False)

    def fake_launch(env, config_path, log_path):
        launches["env"] = env
        launches["config_path"] = config_path

        class DummyProcess:
            pid = 1234

            def poll(self):
                return None

        return DummyProcess()

    monkeypatch.setattr(manager, "_launch_proxy_process", fake_launch)
    monkeypatch.setattr(manager, "_wait_until_ready", lambda timeout=10.0: True)
    monkeypatch.setattr(manager, "_write_state", lambda pid: None)

    info = manager.ensure_started()

    assert info.master_key
    assert info.master_key.startswith("sk-")
    assert launches["env"]["LITELLM_MASTER_KEY"] == info.master_key

    config_data = yaml.safe_load(Path(info.config_path).read_text())
    assert config_data["general_settings"]["master_key"] == info.master_key

    assert "database_url" not in config_data["general_settings"]
