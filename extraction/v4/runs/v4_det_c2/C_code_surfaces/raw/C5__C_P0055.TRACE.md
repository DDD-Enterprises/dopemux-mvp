"""Compatibility shims for legacy migration helpers.

These helpers provide a stable import surface for tests and older callers
that still import from ``dopemux.migration``.
"""

from __future__ import annotations

from dataclasses import dataclass

from .update import UpdateConfig


@dataclass(frozen=True)
class MigrationPlan:
    """Minimal migration plan shape used by compatibility tests/callers."""

    current: str
    target: str
    path: list[str]
    requires_migration: bool
    breaking_changes: bool


def inspect_migration_plan(manager) -> MigrationPlan:
    """Build a normalized migration plan from an update manager."""

    info = manager.check_for_updates()
    return MigrationPlan(
        current=info.current,
        target=info.target,
        path=list(info.migration_path),
        requires_migration=bool(info.requires_migration),
        breaking_changes=bool(info.breaking_changes),
    )


def migration_required(manager) -> bool:
    """Return whether migration is required for the current update path."""

    return inspect_migration_plan(manager).requires_migration


def format_migration_summary(plan: MigrationPlan) -> str:
    """Render a compact summary suitable for logs/CLI output."""

    status = "required" if plan.requires_migration else "not-required"
    breaking = "yes" if plan.breaking_changes else "no"
    return (
        f"current={plan.current} "
        f"target={plan.target} "
        f"migration={status} "
        f"breaking={breaking} "
        f"path={','.join(plan.path)}"
    )


def create_dry_run_update_config(timeout_minutes: int = 30) -> UpdateConfig:
    """Construct an UpdateConfig preset for dry-run migration planning."""

    return UpdateConfig(dry_run=True, timeout_minutes=timeout_minutes)
