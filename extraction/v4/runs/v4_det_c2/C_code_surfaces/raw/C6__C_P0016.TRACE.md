    @pytest.fixture
    def mock_engine(self):
        engine = AsyncMock()

        return engine

    @pytest.fixture
    def voice_assistant(self, mock_engine):
        with patch('subprocess.run') as mock_run:

            va = VoiceAssistant(mock_engine)

            assert args[0] == 'say'
            assert 'Hello world' in args
